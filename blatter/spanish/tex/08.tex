\par \bigskip {\centering\subsection*{Introducción}}

\par \bigskip \noindent El trabajo ${\mathrm{W}}$ realizado por las fuerzas que actúan sobre un sistema de N partículas, la energía cinética ${\mathrm{K}}$ del sistema de partículas, la energía potencial ${\mathrm{U}}$ del sistema de partículas y la energía mecánica ${\mathrm{E}}$ del sistema de partículas, están dados por:

\par \bigskip\medskip ${\mathrm{W}} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}_{cm}) + \Delta \, \med \; {\mathbf{F}}_i \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}_{cm}) \, \big ] \,=\, \Delta \, {\mathrm{K}}$

\par \bigskip\smallskip $\Delta \, {\mathrm{K}} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \Delta \, \med \; m_i \, \big [ \, ({\vec{\mathit{v}}_{i}} - {\vec{\mathit{V}}}_{cm})^2 + ({\vec{\mathit{a}}_{i}} - {\vec{\mathit{A}}}_{cm}) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}_{cm}) \, \big ]$

\par \bigskip\smallskip $\Delta \, {\mathrm{U}} ~\doteq~ - \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}_{cm}) + \Delta \, \med \; {\mathbf{F}}_i \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}_{cm}) \, \big ]$

\par \bigskip\smallskip ${\mathrm{E}} ~\doteq~ {\mathrm{K}} + {\mathrm{U}}$

\par \bigskip\smallskip \noindent donde ${\vec{\mathit{r}}_{i}}$, ${\vec{\mathit{v}}_{i}}$ y ${\vec{\mathit{a}}_{i}}$ son la posición, la velocidad y la aceleración de la \textit{i}-ésima partícula, ${\vec{\mathit{R}}}_{cm}$, ${\vec{\mathit{V}}}_{cm}$ y ${\vec{\mathit{A}}}_{cm}$ son la posición, la velocidad y la aceleración del centro de masa del sistema de partículas, $m_i$ es la masa de la \textit{i}-ésima partícula y ${\mathbf{F}}_i$ es la fuerza neta que actúa sobre la \textit{i}-ésima partícula.

\vspace{+0.30em}

\par \bigskip {\centering\subsection*{Principio}}

\par \bigskip \noindent La energía mecánica ${\mathrm{E}}$ de un sistema de N partículas permanece constante si el sistema está sujeto solamente a fuerzas conservativas.

\par \bigskip\medskip ${\mathrm{E}} ~\doteq~ {\mathrm{K}} + {\mathrm{U}} ~=~ {\mathrm{constante}}$

\par \bigskip\medskip \noindent donde ${\mathrm{K}}$ es la energía cinética del sistema de partículas y ${\mathrm{U}}$ es la energía potencial del sistema de partículas.

\newpage

\par \bigskip {\centering\subsection*{Relaciones}}

\par \bigskip \noindent El trabajo ${\mathrm{W}}$, la energía cinética ${\mathrm{K}}$ y la energía potencial ${\mathrm{U}}$ de un sistema de N partículas (de masa ${\mathrm{M}}$) pueden ser también expresados como sigue:

\par \bigskip \noindent ${\mathrm{W}} ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) + \Delta \, \med \; (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ] ~=~ \Delta \, {\mathrm{K}}$

\par \bigskip \noindent $\Delta \, {\mathrm{K}} ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, ({\vec{\mathit{v}}_{i}} - {\vec{\mathit{v}}_{j}})^2 + ({\vec{\mathit{a}}_{i}} - {\vec{\mathit{a}}_{j}}) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ]$

\par \bigskip \noindent $\Delta \, {\mathrm{U}} ~=~ - \, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) + \Delta \, \med \; (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ]$

\par \bigskip \noindent Si la posición radial ${\mathit{r}}_{\hspace{+0.060em}ij}$, la velocidad radial ${\dot{\mathit{r}}}_{\hspace{+0.060em}ij}$ y la aceleración radial ${\ddot{\mathit{r}}}_{\hspace{+0.060em}ij}$ de un par de partículas $ij$, están dadas por: ${\mathit{r}}_{\hspace{+0.060em}ij} \hspace{-0.060em} \doteq \hspace{-0.060em} | \hspace{+0.090em} {\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j} \hspace{+0.090em} |$, ${\dot{\mathit{r}}}_{\hspace{+0.060em}ij} \doteq d(\hspace{+0.045em}{\mathit{r}}_{\hspace{+0.060em}ij})/dt$ y ${\ddot{\mathit{r}}}_{\hspace{+0.060em}ij} \doteq d^2(\hspace{+0.045em}{\mathit{r}}_{\hspace{+0.060em}ij})/dt^2$ entonces la energía cinética ${\mathrm{K}}$ de un sistema de N partículas puede ser también \hbox {expresada como} sigue:

\par \bigskip \noindent $\Delta \, {\mathrm{K}} ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, {\dot{\mathit{r}}}_{\hspace{+0.060em}ij} \, {\dot{\mathit{r}}}_{\hspace{+0.060em}ij} + {\ddot{\mathit{r}}}_{\hspace{+0.060em}ij} \, {\mathit{r}}_{\hspace{+0.060em}ij} \, \big ]$

\par \bigskip \noindent Si la posición escalar ${\tau}_{\hspace{+0.060em}ij}$, la velocidad escalar ${\dot{\tau}}_{\hspace{+0.060em}ij}$ y la aceleración escalar ${\ddot{\tau}}_{\hspace{+0.060em}ij}$ de un par de partículas $ij$, están dadas por: ${\tau}_{\hspace{+0.060em}ij} \doteq \med \: ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) \cdot ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j})$, ${\dot{\tau}}_{\hspace{+0.060em}ij} \doteq d(\hspace{+0.045em}{\tau}_{\hspace{+0.060em}ij})/dt$ y ${\ddot{\tau}}_{\hspace{+0.060em}ij} \doteq d^2(\hspace{+0.045em}{\tau}_{\hspace{+0.060em}ij})/dt^2$ entonces la energía cinética ${\mathrm{K}}$ de un sistema de N partículas puede ser también expresada como sigue:

\par \bigskip \noindent $\Delta \, {\mathrm{K}} ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, {\ddot{\tau}}_{\hspace{+0.060em}ij} \, \big ]$

\par \bigskip \noindent El trabajo ${\mathrm{W}}$ y la energía potencial ${\mathrm{U}}$ de un sistema aislado de N partículas cuyas fuerzas internas obedecen la tercera ley de Newton en su forma débil se reducen a:

\par \bigskip \noindent ${\mathrm{W}} ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}} + \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}} \, \big ] ~=~ \Delta \, {\mathrm{K}}$

\par \bigskip \noindent $\Delta \, {\mathrm{U}} ~=~ - \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}} + \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}} \, \big ]$

\vspace{+0.60em}

\par \bigskip {\centering\subsection*{Observaciones}}

\par \bigskip \noindent Todas las ecuaciones de este trabajo pueden ser aplicadas en cualquier sistema de referencia inercial o no inercial.

\par \bigskip \noindent Los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias sobre ${\mathbf{F}}_i$ ni sobre ${\mathbf{F}}_j$.

\par \bigskip \noindent En este trabajo, las magnitudes $[ \, {\mathit{m}}, {\vec{\mathit{r}}}_{\hspace{+0.060em}ij}, {\mathit{r}}_{\hspace{+0.060em}ij}, {\dot{\mathit{r}}}_{\hspace{+0.060em}ij}, {\ddot{\mathit{r}}}_{\hspace{+0.060em}ij}, {\tau}_{\hspace{+0.060em}ij}, {\dot{\tau}}_{\hspace{+0.060em}ij}, {\ddot{\tau}}_{\hspace{+0.060em}ij}, {\mathrm{M}}, {\mathbf{F}}, {\mathrm{W}}, {\mathrm{K}}, {\mathrm{U}}$ y ${\mathrm{E}} \; ]$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no \hbox {inerciales}.
