\par \bigskip {\centering\subsection*{Introducción}}\addcontentsline{toc}{subsection}{1. Introducción}

\par \bigskip\smallskip \noindent La nueva teoría, que este trabajo presenta en mecánica relacional, se desarrolla a partir de un sistema auxiliar de partículas (denominado free-system) que es utilizado para obtener magnitudes cinemáticas (denominadas inerciales) que son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \bigskip \noindent La posición inercial ${\mathbf{r}}_i$, la velocidad inercial ${\mathbf{v}}_i$ y la aceleración inercial ${\mathbf{a}}_{\hspace{+0.045em}i}$ de una partícula $i$, están dadas por:

\par \bigskip\smallskip ${\mathbf{r}}_i \,\doteq\, ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip ${\mathbf{v}}_i \,\doteq\, ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip ${\mathbf{a}}_{\hspace{+0.045em}i} \,\doteq\, ({\vec{\mathit{a}}}_i - {\vec{\mathit{A}}}) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}}) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip\smallskip \noindent $( \, {\mathbf{v}}_i \doteq d(\hspace{+0.090em}{\mathbf{r}}_i\hspace{+0.090em})/dt \, )$ y $( \, {\mathbf{a}}_{\hspace{+0.045em}i} \doteq d^2(\hspace{+0.090em}{\mathbf{r}}_i\hspace{+0.090em})/dt^2 \, )$ donde ${\vec{\mathit{r}}}_i$ es el vector de posición de la partícula $i$, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del free-system y ${\vec{\omega}}$ es el vector de velocidad angular del free-system \hyperlink{t3a1}{(ver Anexo I\hspace{+0.120em})}

\par \bigskip \noindent Un sistema de referencia S es no rotante si la velocidad angular ${\vec{\omega}}$ del free-system \hbox {respecto a S} es igual a cero y S es además inercial si la aceleración ${\vec{\mathit{A}}}$ del centro de masa del free-system respecto a S es igual a cero.

\vspace{+1.20em}

\par {\centering\subsection*{Nueva Dinámica}}\addcontentsline{toc}{subsection}{2. Nueva Dinámica}

\par \bigskip\smallskip \noindent $[\,1\,]$ Toda fuerza siempre es causada por la interacción entre dos o más partículas.

\par \bigskip \noindent $[\,2\,]$ La fuerza neta ${\mathbf{F}}_i$ que actúa sobre una partícula $i$ de masa $m_i$ produce una aceleración inercial ${\mathbf{a}}_{\hspace{+0.045em}i}$, según la siguiente ecuación: $[ \, {\mathbf{F}}_i \,=\, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} \, ]$

\par \bigskip \noindent $[\,3\,]$ En este trabajo, todas las fuerzas pueden obedecer o desobedecer la tercera ley de Newton en su forma débil y{\small /}o en su forma fuerte.

\par {\centering\subsection*{Relaciones}}\addcontentsline{toc}{subsection}{4. Relaciones}

\par \bigskip\smallskip \noindent En un sistema de partículas, entre las energías cinéticas, las energías potenciales y las energías mecánicas, se dan siempre estas relaciones \hyperlink{t3a2}{(ver Anexo II\hspace{+0.120em})}

\vspace{+1.20em}

\par \noindent ${\mathrm{K}}_1 ~=~ {\mathrm{K}}_2 + \med \; {\mathrm{M}} \: {\mathbf{V}}_{cm}^{\hspace{+0.045em}2}$
\vspace{+0.99em}
\par \noindent ${\mathrm{K}}_3 ~=~ {\mathrm{K}}_4 + \med \; {\mathrm{M}} \: {\mathbf{A}}_{cm} \cdot {\mathbf{R}}_{cm}$
\vspace{+0.99em}
\par \noindent ${\mathrm{K}}_5 ~=~ {\mathrm{K}}_6 + \med \; {\mathrm{M}} \: \big [ \, ({\vec{\mathit{V}}}_{cm} - \hspace{-0.120em}{\vec{\mathit{V}}})^2 + ({\vec{\mathit{A}}}_{cm} - {\vec{\mathit{A}}}) \cdot ({\vec{\mathit{R}}}_{cm} - {\vec{\mathit{R}}}) \, \big ]$
\vspace{+0.99em}
\par \noindent ${\mathrm{K}}_5 ~=~ {\mathrm{K}}_1 + {\mathrm{K}}_3$ $\hspace{+0.540em} \& \hspace{+0.540em}$ ${\mathrm{U}}_5 ~=~ {\mathrm{U}}_1 \hspace{+0.027em}+\hspace{+0.027em} {\mathrm{U}}_3$ $\hspace{+0.630em} \& \hspace{+0.630em}$ ${\mathrm{E}}_5 ~=~ {\mathrm{E}}_1 + {\mathrm{E}}_3$
\vspace{+0.99em}
\par \noindent ${\mathrm{K}}_6 ~=~ {\mathrm{K}}_2 + {\mathrm{K}}_4$ $\hspace{+0.540em} \& \hspace{+0.540em}$ ${\mathrm{U}}_6 ~=~ {\mathrm{U}}_2 \hspace{+0.027em}+\hspace{+0.027em} {\mathrm{U}}_4$ $\hspace{+0.630em} \& \hspace{+0.630em}$ ${\mathrm{E}}_6 ~=~ {\mathrm{E}}_2 + {\mathrm{E}}_4$

\par {\centering\subsection*{Principios}}\addcontentsline{toc}{subsection}{5. Principios}

\par \bigskip\smallskip \noindent El momento lineal $[ \, {\mathbf{P}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas internas obedecen la tercera ley de Newton en su forma débil.

\par \bigskip\medskip ${\mathbf{P}}_1 ~=~ {\mathrm{constante}} \hspace{+2.88em} \big [ \; d({\mathbf{P}}_1)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent El momento angular $[ \, {\mathbf{L}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas internas obedecen la tercera ley de Newton en su forma fuerte.

\par \bigskip\medskip ${\mathbf{L}}_1 ~=~ {\mathrm{constante}} \hspace{+2.97em} \big [ \; d({\mathbf{L}}_1)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, {\mathbf{r}}_i \times {\mathbf{a}}_{\hspace{+0.045em}i} \, \big ]~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, {\mathbf{r}}_i \times {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent El momento angular $[ \, {\mathbf{L}}_2 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas internas obedecen la tercera ley de Newton en su forma fuerte.

\par \bigskip\medskip ${\mathbf{L}}_2 ~=~ {\mathrm{constante}} \hspace{+2.97em} \big [ \; d({\mathbf{L}}_2)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times ({\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{A}}_{cm}) \, \big ] ~=~$

\par \bigskip $\hspace{+10.44em} \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times {\mathbf{a}}_{\hspace{+0.045em}i} \, \big ] ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_1$ y ${\mathrm{E}}_2 \, ]$ de un sistema de N partículas permanecen constantes si el sistema está sujeto solamente a fuerzas conservativas.

\par \bigskip\medskip ${\mathrm{E}}_1 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_1 ~=~ \Delta \; {\mathrm{K}}_1 + \Delta \; {\mathrm{U}}_1 ~=~ 0 \; \big ]$

\par \bigskip ${\mathrm{E}}_2 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_2 ~=~ \Delta \; {\mathrm{K}}_2 + \Delta \; {\mathrm{U}}_2 ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_3$ y ${\mathrm{E}}_4 \, ]$ de un sistema de N partículas son siempre iguales a cero, por lo tanto, permanecen siempre constantes.

\par \bigskip\medskip ${\mathrm{E}}_3 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; {\mathrm{E}}_3 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; \big [ \, m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot {\mathbf{r}}_{i} - {\mathbf{F}}_i \cdot {\mathbf{r}}_{i} \, \big ] ~=~ 0 \; \big ]$

\par \bigskip ${\mathrm{E}}_4 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; {\mathrm{E}}_4 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; \big [ \, m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm}) - {\mathbf{F}}_i \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm}) \, \big ] ~=~ 0 \; \big ]$

\par \bigskip $\hspace{+10.44em} \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, \big [ \, ({\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{A}}_{cm}) \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm}) \, \big ] \,=\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm})$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_5$ y ${\mathrm{E}}_6 \, ]$ de un sistema de N partículas permanecen constantes si el sistema está sujeto solamente a fuerzas conservativas.

\par \bigskip\medskip ${\mathrm{E}}_5 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_5 ~=~ \Delta \; {\mathrm{K}}_5 + \Delta \; {\mathrm{U}}_5 ~=~ 0 \; \big ]$

\par \bigskip ${\mathrm{E}}_6 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_6 ~=~ \Delta \; {\mathrm{K}}_6 + \Delta \; {\mathrm{U}}_6 ~=~ 0 \; \big ]$

\par \bigskip {\centering\subsection*{Observaciones}}\addcontentsline{toc}{subsection}{6. Observaciones}

\par \bigskip\smallskip \noindent Todas las ecuaciones de este trabajo pueden ser aplicadas en cualquier sistema de referencia inercial o no inercial.

\par \bigskip\smallskip \noindent Los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias sobre ${\mathbf{F}}_i$.

\par \bigskip\smallskip \noindent En este trabajo, las magnitudes $[ \, {\mathit{m}},\, {\mathbf{r}},\, {\mathbf{v}},\, {\mathbf{a}},\, {\mathrm{M}},\, {\mathbf{R}}, {\mathbf{V}}, {\mathbf{A}},\, {\mathbf{F}},\, {\mathbf{P}}_1, {\mathbf{L}}_1, {\mathbf{L}}_2, {\mathrm{W}}_1, {\mathrm{K}}_1, {\mathrm{U}}_1, {\mathrm{E}}_1, {\mathrm{L}}_1$, ${\mathrm{W}}_2,\spc {\mathrm{K}}_2,\spc {\mathrm{U}}_2,\spc {\mathrm{E}}_2,\spc {\mathrm{L}}_2,\spc {\mathrm{W}}_3,\spc {\mathrm{K}}_3,\spc {\mathrm{U}}_3,\spc {\mathrm{E}}_3,\spc {\mathrm{W}}_4,\spc {\mathrm{K}}_4,\spc {\mathrm{U}}_4,\spc {\mathrm{E}}_4,\spc {\mathrm{W}}_5,\spc {\mathrm{K}}_5,\spc {\mathrm{U}}_5,\spc {\mathrm{E}}_5,\spc {\mathrm{W}}_6,\spc {\mathrm{K}}_6,\spc {\mathrm{U}}_6$ y ${\mathrm{E}}_6 \, ]$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no \hbox {inerciales}.

\par \bigskip\smallskip \noindent La energía mecánica ${\mathrm{E}}_3$ de un sistema de partículas es siempre igual a cero $[ \, {\mathrm{E}}_3 = {\mathrm{K}}_3 + {\mathrm{U}}_3 = 0 \, ]$

\par \bigskip\smallskip \noindent Por lo tanto, la energía mecánica ${\mathrm{E}}_5$ de un sistema de partículas es siempre igual a la energía mecánica ${\mathrm{E}}_1$ del sistema de partículas $[ \, {\mathrm{E}}_5 = {\mathrm{E}}_1 \, ]$

\par \bigskip\smallskip \noindent La energía mecánica ${\mathrm{E}}_4$ de un sistema de partículas es siempre igual a cero $[ \, {\mathrm{E}}_4 = {\mathrm{K}}_4 + {\mathrm{U}}_4 = 0 \, ]$

\par \bigskip\smallskip \noindent Por lo tanto, la energía mecánica ${\mathrm{E}}_6$ de un sistema de partículas es siempre igual a la energía mecánica ${\mathrm{E}}_2$ del sistema de partículas $[ \, {\mathrm{E}}_6 = {\mathrm{E}}_2 \, ]$

\par \bigskip\smallskip \noindent Si la energía potencial ${\mathrm{U}}_1$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} entonces la energía potencial ${\mathrm{U}}_3$ y la energía potencial ${\mathrm{U}}_5$ del sistema de partículas, están dadas por: $[ \, {\mathrm{U}}_3 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 \, ]$ y $[ \, {\mathrm{U}}_5 = ({\scriptstyle 1 +} \frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 \, ]$

\par \bigskip\smallskip \noindent Si la energía potencial ${\mathrm{U}}_2$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} entonces la energía potencial ${\mathrm{U}}_4$ y la energía potencial ${\mathrm{U}}_6$ del sistema de partículas, están dadas por: $[ \, {\mathrm{U}}_4 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 \, ]$ y $[ \, {\mathrm{U}}_6 = ({\scriptstyle 1 +} \frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 \, ]$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_4 ~=~ - \, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot ({\mathbf{r}}_{i} - {\mathbf{r}}_{j}) \, \big ]$

\par \bigskip\bigskip \noindent ${\mathrm{W}}_6 ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) + \Delta \, \med \; ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ]$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{K}}_6 ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, ({\vec{\mathit{v}}_{i}} - {\vec{\mathit{v}}_{j}})^2 + ({\vec{\mathit{a}}_{i}} - {\vec{\mathit{a}}_{j}}) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ] ~=~ {\mathrm{W}}_6$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_6 ~=~ - \, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) + \Delta \, \med \; ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ]$

\par \bigskip\bigskip \noindent Las magnitudes ${\mathrm{W}}_{(1\;{\mathrm{al}}\;6)}$ y ${\mathrm{U}}_{(1\;{\mathrm{al}}\;6)}$ de un sistema aislado de N partículas cuyas fuerzas internas obedecen la tercera ley de Newton en su forma débil se reducen a:

\par \bigskip\bigskip \noindent ${\mathrm{W}}_1 ~=~ {\mathrm{W}}_2 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}}$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_1 ~=~ \Delta \, {\mathrm{U}}_2 ~=~ - \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}}$

\par \bigskip\bigskip \noindent ${\mathrm{W}}_3 ~=~ {\mathrm{W}}_4 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}}$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_3 ~=~ \Delta \, {\mathrm{U}}_4 ~=~ - \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}}$

\par \bigskip\bigskip \noindent ${\mathrm{W}}_5 ~=~ {\mathrm{W}}_6 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}} + \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}} \, \big ]$

\par \bigskip {\centering\subsection*{Introducción}}\addcontentsline{toc}{subsection}{1. Introducción}

\par \bigskip\smallskip \noindent La nueva teoría, que este trabajo presenta en mecánica relacional, se desarrolla a partir de un sistema auxiliar de partículas (denominado free-system) que es utilizado para obtener magnitudes cinemáticas (denominadas inerciales) que son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \bigskip \noindent La posición inercial ${\mathbf{r}}_i$, la velocidad inercial ${\mathbf{v}}_i$ y la aceleración inercial ${\mathbf{a}}_{\hspace{+0.045em}i}$ de una partícula $i$, están dadas por:

\par \bigskip\smallskip ${\mathbf{r}}_i \,\doteq\, ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip ${\mathbf{v}}_i \,\doteq\, ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip ${\mathbf{a}}_{\hspace{+0.045em}i} \,\doteq\, ({\vec{\mathit{a}}}_i - {\vec{\mathit{A}}}) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}}) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip\smallskip \noindent $( \, {\mathbf{v}}_i \doteq d(\hspace{+0.090em}{\mathbf{r}}_i\hspace{+0.090em})/dt \, )$ y $( \, {\mathbf{a}}_{\hspace{+0.045em}i} \doteq d^2(\hspace{+0.090em}{\mathbf{r}}_i\hspace{+0.090em})/dt^2 \, )$ donde ${\vec{\mathit{r}}}_i$ es el vector de posición de la partícula $i$, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del free-system y ${\vec{\omega}}$ es el vector de velocidad angular del free-system \hyperlink{t4a1}{(ver Anexo I\hspace{+0.120em})}

\par \bigskip \noindent Un sistema de referencia S es no rotante si la velocidad angular ${\vec{\omega}}$ del free-system \hbox {respecto a S} es igual a cero y S es además inercial si la aceleración ${\vec{\mathit{A}}}$ del centro de masa del free-system respecto a S es igual a cero.

\vspace{+1.20em}

\par {\centering\subsection*{Nueva Dinámica}}\addcontentsline{toc}{subsection}{2. Nueva Dinámica}

\par \bigskip\smallskip \noindent $[\,1\,]$ Toda fuerza siempre es causada por la interacción entre dos o más partículas.

\par \bigskip \noindent $[\,2\,]$ La fuerza total ${\mathbf{T}}_i$ que actúa sobre una partícula $i$ es siempre igual a cero $[ \, {\mathbf{T}}_i \,=\, 0 \, ]$

\par \bigskip \noindent $[\,3\,]$ En este trabajo, todas las fuerzas no cinéticas pueden obedecer o desobedecer la tercera ley \hbox {de Newton} en su forma débil y{\small /}o en su forma fuerte.

\newpage

\par \bigskip {\centering\subsection*{Fuerzas Cinéticas}}\addcontentsline{toc}{subsection}{3. Fuerzas Cinéticas}

\par \bigskip\smallskip \noindent La fuerza cinética ${\mathbf{K}}^{a}_{\hspace{+0.060em}ij}$ ejercida sobre una partícula $i$ de masa $m_i$ por otra partícula $j$ de masa $m_j$, causada por la interacción entre la partícula $i$ y la partícula $j$, está dada por:

\par \bigskip ${\mathbf{K}}^{a}_{\hspace{+0.060em}ij} ~=~ - \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \, (\hspace{+0.045em}{\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{a}}_{j})$

\par \bigskip \noindent donde ${\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración inercial de la partícula $i$, ${\mathbf{a}}_{j}$ es la aceleración inercial de la \hbox {partícula $j$} y ${\mathrm{M}}$ es la masa del Universo.

\par \bigskip \noindent La fuerza cinética ${\mathbf{K}}^{u}_{i}$ ejercida sobre una partícula $i$ de masa $m_i$ por el centro de masa del Universo, causada por la interacción entre la partícula $i$ y el Universo, está dada por:

\par \bigskip ${\mathbf{K}}^{u}_{i} ~=~ - \; m_i \, {\mathbf{A}}_{cm}$

\par \bigskip \noindent donde ${\mathbf{A}}_{cm}$ es la aceleración inercial del centro de masa del Universo.

\par \bigskip \noindent De las ecuaciones anteriores se deduce que la fuerza cinética neta ${\mathbf{K}}_{i}$ \hbox {(\hspace{+0.240em}$ = \sum_{j}^{\scriptscriptstyle{\mathit{All}}} \, {\mathbf{K}}^{a}_{\hspace{+0.060em}ij} + {\mathbf{K}}^{u}_{i}$\hspace{+0.240em})} que actúa sobre una partícula $i$ de masa $m_i$, está dada por:

\par \bigskip ${\mathbf{K}}_{i} ~=~ - \; m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent donde ${\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración inercial de la partícula $i$.

\par \bigskip \noindent Las fuerzas cinéticas ${\mathbf{K}}^{a}$ obedecen siempre la tercera ley de Newton en su forma débil.

\par \bigskip \noindent Si todas las fuerzas no cinéticas obedecen siempre la tercera ley de Newton en su forma débil entonces la aceleración inercial del centro de masa del Universo ${\mathbf{A}}_{cm}$ es siempre \hbox {igual a cero}.

\vspace{+1.65em}

\par {\centering\subsection*{[$\,$2$\,$] Principio}}\addcontentsline{toc}{subsection}{4. Segundo Principio}

\par \bigskip\smallskip \noindent El $[\,2\,]$ principio de la nueva dinámica establece que la fuerza total ${\mathbf{T}}_i$ que actúa sobre una partícula $i$ es siempre igual a cero.

\par \bigskip ${\mathbf{T}}_i ~=~ 0$

\par \bigskip \noindent Si la fuerza total ${\mathbf{T}}_i$ es dividida en las siguientes dos partes: la fuerza cinética neta ${\mathbf{K}}_{i}$ y la fuerza no cinética neta ${\mathbf{F}}_{i}$ (\hspace{+0.060em}$\sum$ de fuerzas gravitatorias, fuerzas electrostáticas, etc.\hspace{+0.060em}) entonces:

\par \bigskip ${\mathbf{K}}_{i} + {\mathbf{F}}_{i} ~=~ 0$

\par \bigskip \noindent Ahora, sustituyendo (\hspace{+0.060em}${\mathbf{K}}_{i} \,=\, - \; m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$\hspace{+0.060em}) y reordenando, finalmente se obtiene:

\par \bigskip ${\mathbf{F}}_{i} ~=~ m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent Esta ecuación (\hspace{+0.060em}similar a la segunda ley de Newton\hspace{+0.060em}) será usada a lo largo de este trabajo.

\par \bigskip \noindent Por otro lado, en este trabajo un sistema de partículas es aislado cuando el sistema está libre de fuerzas no cinéticas externas.

\par {\centering\subsection*{Principios}}\addcontentsline{toc}{subsection}{7. Principios}

\par \bigskip\smallskip \noindent El momento lineal $[ \, {\mathbf{P}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas no cinéticas internas obedecen la tercera ley de Newton en su forma débil.

\par \bigskip\medskip ${\mathbf{P}}_1 ~=~ {\mathrm{constante}} \hspace{+2.88em} \big [ \; d({\mathbf{P}}_1)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent El momento angular $[ \, {\mathbf{L}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas no cinéticas internas obedecen la tercera ley de Newton en su forma fuerte.
