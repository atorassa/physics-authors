\par \bigskip {\centering\subsection*{Introducción}}

\par \bigskip \noindent La posición inercial ${\mathbf{r}}$, la velocidad inercial ${\mathbf{v}}$ y la aceleración inercial ${\mathbf{a}}$ de una partícula, están dadas por:

\par \bigskip \hspace{+0.75em} ${\mathbf{r}} \,\doteq\, ({\vec{\mathit{r}}} - {\vec{\mathit{R}}})$

\par \bigskip \hspace{+0.75em} ${\mathbf{v}} \,\doteq\, ({\vec{\mathit{v}}} - {\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}} - {\vec{\mathit{R}}})$

\par \bigskip \hspace{+0.75em} ${\mathbf{a}} \,\doteq\, ({\vec{\mathit{a}}} - {\vec{\mathit{A}}}) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}} - {\vec{\mathit{V}}}) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}} - {\vec{\mathit{R}}}) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}} - {\vec{\mathit{R}}})$

\par \bigskip \noindent {\small $( \, {\mathbf{v}} \doteq d(\hspace{+0.090em}{\mathbf{r}}\hspace{+0.090em})/dt \, )$} y {\small $( \, {\mathbf{a}} \doteq d^2(\hspace{+0.090em}{\mathbf{r}}\hspace{+0.090em})/dt^2 \, )$} donde ${\vec{\mathit{r}}}$ es el vector de posición de la partícula, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del free-system y ${\vec{\omega}}$ es el vector de velocidad angular del free-system (ver Anexo I\hspace{+0.120em})

\par \bigskip \noindent El momento inercial ${\mathbf{P}}$ de una partícula, está dado por:

\par \bigskip \hspace{+0.75em} ${\mathbf{P}} \,\doteq\, m \, {\mathbf{v}}$

\par \bigskip \noindent donde $m$ es la masa de la partícula y ${\mathbf{v}}$ es la velocidad inercial de la partícula.

\par \bigskip \noindent La fuerza neta ${\mathbf{F}}$ que actúa sobre una partícula $m$ produce una aceleración inercial ${\mathbf{a}}$, según la siguiente ecuación:

\par \bigskip \hspace{+0.75em} ${\mathbf{F}} \,=\, m \, {\mathbf{a}}$

\par \bigskip \noindent Los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias sobre ${\mathbf{F}}$.

\par \bigskip \noindent Las magnitudes ${\mathit{m}}$, ${\mathbf{r}}$, ${\mathbf{v}}$, ${\mathbf{a}}$, ${\mathbf{P}}$ y ${\mathbf{F}}$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \bigskip\medskip {\centering\subsubsection*{W, K, U, E y L}}

\par \bigskip \noindent En este trabajo, el trabajo ${\mathrm{W}}$ realizado por la fuerza neta ${\mathbf{F}}$ que actúa sobre una partícula, está dado por:

\par \bigskip \hspace{+0.75em} ${\mathrm{W}} \,\doteq\, \int_1^2 \hspace{+0.045em} {\mathbf{F}} \cdot d\hspace{+0.090em}(\hspace{+0.090em}{\mathbf{r}}\hspace{+0.090em}) \,=\, \Delta \, \med \; m \, ({\mathbf{v}})^2$

\par \bigskip \noindent donde ${\mathbf{r}}$ es la posición inercial de la partícula, $m$ es la masa de la partícula y ${\mathbf{v}}$ es la velocidad inercial de la partícula.

\par \bigskip \noindent La energía cinética ${\mathrm{K}}$ de una partícula, está dada por:

\par \bigskip \hspace{+0.75em} ${\mathrm{K}} \,\doteq\, \med \; m \, ({\mathbf{v}})^2$

\par \bigskip \noindent donde $m$ es la masa de la partícula y ${\mathbf{v}}$ es la velocidad inercial de la partícula.

\par \bigskip \noindent Por lo tanto, la energía cinética ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}}$ de un sistema de N partículas, está dada por:

\par \bigskip \hspace{+0.75em} ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}} \,\doteq\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, ({\mathbf{v}}_i)^2$

\par \bigskip \noindent donde $m_i$ es la masa de la \textit{i}-ésima partícula y ${\mathbf{v}}_i$ es la velocidad inercial de la \textit{i}-ésima partícula.

\par \bigskip \noindent El trabajo ${\mathrm{W}}$ realizado por las fuerzas conservativas ${\mathbf{F}}$ que actúan sobre una partícula es igual y de signo opuesto a la variación en la energía potencial ${\mathrm{U}}$ de la partícula.

\par \bigskip \hspace{+0.75em} $\Delta \, {\mathrm{U}} \,\doteq\, - \, \int_1^2 \hspace{+0.045em} {\mathbf{F}} \cdot d\hspace{+0.090em}(\hspace{+0.090em}{\mathbf{r}}\hspace{+0.090em})$

\par \bigskip \noindent donde ${\mathbf{r}}$ es la posición inercial de la partícula.

\par \bigskip \noindent Por lo tanto, la energía mecánica ${\mathrm{E}}$ de una partícula permanece constante si la partícula está sujeta sólo a fuerzas conservativas. 

\par \bigskip \hspace{+0.75em} $\Delta \, {\mathrm{E}} \,\doteq\, \Delta \, {\mathrm{K}} + \Delta \, {\mathrm{U}} \,=\, 0$

\par \bigskip \hspace{+0.75em} ${\mathrm{E}} \,\doteq\, {\mathrm{K}} + {\mathrm{U}} \,=\, {\mathrm{constante}}$

\par \bigskip \noindent donde ${\mathrm{K}}$ es la energía cinética de la partícula y ${\mathrm{U}}$ es la energía potencial de la partícula.

\par \bigskip \noindent Finalmente, el Lagrangiano ${\mathrm{L}}$ de una partícula, está dado por:

\par \bigskip \hspace{+0.75em} ${\mathrm{L}} \,\doteq\, {\mathrm{K}} - {\mathrm{U}}$

\par \bigskip \noindent donde ${\mathrm{K}}$ es la energía cinética de la partícula y ${\mathrm{U}}$ es la energía potencial de la partícula.

\par {\centering\subsubsection*{Definiciones}}

\par \bigskip\smallskip ${\mathrm{M}} \,\doteq\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i$

\par \bigskip\smallskip ${\mathbf{V}}_{cm} \,\doteq\, {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{v}}_i$

\par \bigskip\smallskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{L}}} \,\doteq\, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, m_j \, {\mathrm M^{\scriptscriptstyle -1}} \, [ \, ({\mathbf{v}}_i - {\mathbf{v}}_j) \, ]^2$

\par \bigskip\smallskip ${\hphantom{{\mathrm{K}}^{\scriptscriptstyle{\mathrm{L}}}}} \,=\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, [ \, ({\mathbf{v}}_i - {\mathbf{V}}_{cm}) \, ]^2$

\par \bigskip\smallskip ${\hphantom{{\mathrm{K}}^{\scriptscriptstyle{\mathrm{L}}}}} \,=\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, ({\mathbf{v}}_i)^2 - \med \; {\mathrm{M}} \, {\mathbf{V}}_{cm}{\hphantom{{\mathbf{V}}}}^{\hspace{-1.92em}2}$

\par \bigskip\smallskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}} \,\doteq\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, ({\mathbf{v}}_i)^2$

\par \bigskip\smallskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{P}}} \,\doteq\, \med \; {\mathrm{M}} \, {\mathbf{V}}_{cm}{\hphantom{{\mathbf{V}}}}^{\hspace{-1.92em}2}$

\par \bigskip\smallskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{A}}} \,\doteq\, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, m_j \, {\mathrm M^{\scriptscriptstyle -1}} \, [ \, ({\mathbf{r}}_i - {\mathbf{r}}_j) \times ({\mathbf{v}}_i - {\mathbf{v}}_j) \hspace{+0.120em}/\hspace{+0.120em} |\hspace{+0.090em}{\mathbf{r}}_i - {\mathbf{r}}_j\hspace{+0.045em}| \: ]^2$

\par \bigskip\smallskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{R}}} \,\doteq\, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, m_j \, {\mathrm M^{\scriptscriptstyle -1}} \, [ \, ({\mathbf{r}}_i - {\mathbf{r}}_j) \cdot ({\mathbf{v}}_i - {\mathbf{v}}_j) \hspace{+0.120em}/\hspace{+0.120em} |\hspace{+0.090em}{\mathbf{r}}_i - {\mathbf{r}}_j\hspace{+0.045em}| \: ]^2$

\vspace{+1.80em}

\par {\centering\subsubsection*{Relaciones}}

\par \bigskip\smallskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{L}}} \,=\, {\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}} - {\mathrm{K}}^{\scriptscriptstyle{\mathrm{P}}}$

\par \bigskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{L}}} \,=\, {\mathrm{K}}^{\scriptscriptstyle{\mathrm{A}}} + {\mathrm{K}}^{\scriptscriptstyle{\mathrm{R}}}$

\par \bigskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}} \,=\, {\mathrm{K}}^{\scriptscriptstyle{\mathrm{P}}} + {\mathrm{K}}^{\scriptscriptstyle{\mathrm{A}}} + {\mathrm{K}}^{\scriptscriptstyle{\mathrm{R}}}$

\par \bigskip\bigskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{L}}} \,=\,$ La energía cinética lineal del sistema de partículas.

\par \bigskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}} \,=\,$ La energía cinética del sistema de partículas.

\par \bigskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{P}}} \,=\,$ La energía cinética del centro de masa del sistema de partículas.

\par \bigskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{A}}} \,=\,$ La energía cinética angular del sistema de partículas.

\par \bigskip ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{R}}} \,=\,$ La energía cinética radial del sistema de partículas.

\par \bigskip {\centering\subsection*{Observaciones}}

\par \bigskip\smallskip \noindent La energía cinética ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}}$ de un sistema de partículas puede ser separada en tres partes:

\par \bigskip \noindent ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{P}}} \,=\,$ La energía cinética del centro de masa del sistema de partículas.

\par \bigskip \noindent ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{A}}} \,=\,$ La energía cinética angular del sistema de partículas.

\par \bigskip \noindent ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{R}}} \,=\,$ La energía cinética radial del sistema de partículas.

\par \bigskip \noindent La energía cinética ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}}$ de un sistema de partículas puede ser representada por un par de partículas iguales (ver Anexo II\hspace{+0.120em}) en el cual:

\par \bigskip \noindent 1\hspace{+0.120em}) La masa del sistema de partículas es representada por la masa del par de partículas.

\par \bigskip \noindent 2\hspace{+0.120em}) El centro de masa del sistema de partículas es representado por el centro de masa del par de partículas.

\par \bigskip \noindent 3\hspace{+0.120em}) El momento de inercia del sistema de partículas es representado por el momento de inercia del par de partículas.

\par \bigskip \noindent 4\hspace{+0.120em}) La energía cinética del centro de masa del sistema de partículas es representada por la energía cinética del centro de masa del par de partículas.

\par \bigskip \noindent 5\hspace{+0.120em}) La energía cinética angular del sistema de partículas es representada por la energía cinética angular del par de partículas.

\par \bigskip \noindent 6\hspace{+0.120em}) La energía cinética radial del sistema de partículas es representada por la energía cinética radial del par de partículas.

\par \bigskip \noindent Por otro lado, todas las ecuaciones de este trabajo pueden ser aplicadas en cualquier sistema de referencia inercial o no inercial.

\par \bigskip \noindent Los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias sobre ${\mathbf{F}}$.

\par \bigskip \noindent En este trabajo, las magnitudes $[ \, {\mathit{m}}, {\mathbf{r}}, {\mathbf{v}}, {\mathbf{a}}, {\mathbf{P}}, {\mathbf{F}}, {\mathrm{W}}, {\mathrm{K}}, {\mathrm{U}}, {\mathrm{E}}$ y ${\mathrm{L}} \, ]$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \bigskip \noindent Un sistema de referencia S es no rotante si la velocidad angular ${\vec{\omega}}$ del free-system respecto a S es igual a cero y además S es inercial si la aceleración ${\vec{\mathit{A}}}$ del centro de masa del free-system respecto a S es igual a cero.

\par \bigskip \noindent Este trabajo no contradice la dinámica de Newton. De hecho, la ecuación $[ \, {\mathbf{F}} \,=\, m \, {\mathbf{a}} \, ]$ es una simple reformulación de la segunda ley de Newton.

\par \bigskip \noindent Finalmente, este trabajo considera que es posible desarrollar una dinámica clásica \hbox {alternativa} basada en el Lagrangiano ${\mathrm{L}}$ que puede ser aplicada en cualquier sistema de referencia inercial o no inercial sin necesidad de introducir las fuerzas ficticias.

\par \bigskip {\centering\subsection*{Anexo I}}

\par \medskip {\centering\subsubsection*{Free-System}}

\par \bigskip \noindent El free-system es un sistema de N partículas que está siempre libre de fuerzas externas e internas, que es tridimensional y que las distancias relativas entre las N partículas permanecen siempre constantes.

\par \bigskip \noindent La posición ${\vec{\mathit{R}}}$, la velocidad ${\vec{\mathit{V}}}$ y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del free-system respecto a un sistema de referencia S, la velocidad angular ${\vec{\omega}}$ y la aceleración \hbox {angular ${\vec{\alpha}}$} del free-system respecto al sistema de referencia S, están dadas por:

\par \bigskip\smallskip \hspace{-2.40em} \begin{tabular}{l}
${\mathrm{M}} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i$ \vspace{+1.20em} \\
${\vec{\mathit{R}}} ~\doteq~ {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\vec{\mathit{r}}_{i}}$ \vspace{+1.20em} \\
${\vec{\mathit{V}}} ~\doteq~ {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\vec{\mathit{v}}_{i}}$ \vspace{+1.20em} \\
${\vec{\mathit{A}}} ~\doteq~ {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\vec{\mathit{a}}_{i}}$ \vspace{+1.20em} \\
${\vec{\omega}} ~\doteq~ {\mathit{I}}^{\scriptscriptstyle -1}{\vphantom{\sum_1^2}}^{\hspace{-1.500em}\leftrightarrow}\hspace{+0.600em} \cdot {\vec{\mathit{L}}}$ \vspace{+1.20em} \\
${\vec{\alpha}} ~\doteq~ d({\vec{\omega}})/dt$ \vspace{+1.20em} \\
${\mathit{I}}{\vphantom{\sum_1^2}}^{\hspace{-0.555em}\leftrightarrow}\hspace{-0.210em} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, [ \, |\hspace{+0.090em}{\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}\,|^2 \hspace{+0.309em} {\mathrm{1}}{\vphantom{\sum_1^2}}^{\hspace{-0.639em}\leftrightarrow}\hspace{-0.129em} - ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}) \otimes ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}) \, ]$ \vspace{+1.20em} \\
${\vec{\mathit{L}}} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}) \times ({\vec{\mathit{v}}_{i}} - {\vec{\mathit{V}}})$
\end{tabular}

\par \bigskip \noindent donde ${\mathrm{M}}$ es la masa del free-system, ${\mathit{I}}{\vphantom{\sum_1^2}}^{\hspace{-0.555em}\leftrightarrow}\hspace{-0.300em}$ es el tensor de inercia del free-system (respecto a ${\vec{\mathit{R}}}$) y \hbox {${\vec{\mathit{L}}}$ es el momento} angular del free-system respecto al sistema de referencia S.

\vspace{+1.50em}

\par {\centering\subsubsection*{Transformaciones}}

\par \bigskip\medskip \hspace{-1.80em} $({\vec{\mathit{r}}} - {\vec{\mathit{R}}}) ~\doteq~ {\mathbf{r}} ~=~ {\mathbf{r}}\hspace{+0.060em}'$

\par \bigskip \hspace{-1.80em} $({\vec{\mathit{r}}}\hspace{+0.060em}' - {\vec{\mathit{R}}}\hspace{+0.060em}') ~\doteq~ {\mathbf{r}}\hspace{+0.060em}' ~=~ {\mathbf{r}}$

\par \bigskip \hspace{-1.80em} $({\vec{\mathit{v}}} - {\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}} - {\vec{\mathit{R}}}) ~\doteq~ {\mathbf{v}} ~=~ {\mathbf{v}}\hspace{+0.060em}'$

\par \bigskip \hspace{-1.80em} $({\vec{\mathit{v}}}\hspace{+0.060em}' - {\vec{\mathit{V}}}\hspace{+0.060em}') - {\vec{\omega}}\hspace{+0.060em}' \times ({\vec{\mathit{r}}}\hspace{+0.060em}' - {\vec{\mathit{R}}}\hspace{+0.060em}') ~\doteq~ {\mathbf{v}}\hspace{+0.060em}' ~=~ {\mathbf{v}}$

\par \bigskip \hspace{-1.80em} $({\vec{\mathit{a}}} - {\vec{\mathit{A}}}) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}} - {\vec{\mathit{V}}}) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}} - {\vec{\mathit{R}}}) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}} - {\vec{\mathit{R}}}) ~\doteq~ {\mathbf{a}} ~=~ {\mathbf{a}}\hspace{+0.060em}'$

\par \bigskip \hspace{-1.80em} $({\vec{\mathit{a}}}\hspace{+0.060em}' - {\vec{\mathit{A}}}\hspace{+0.060em}') - 2 \; {\vec{\omega}}\hspace{+0.060em}' \times ({\vec{\mathit{v}}}\hspace{+0.060em}' - {\vec{\mathit{V}}}\hspace{+0.060em}') + {\vec{\omega}}\hspace{+0.060em}' \times [ \, {\vec{\omega}}\hspace{+0.060em}' \times ({\vec{\mathit{r}}}\hspace{+0.060em}' - {\vec{\mathit{R}}}\hspace{+0.060em}') \, ] - {\vec{\alpha}}\hspace{+0.060em}' \times ({\vec{\mathit{r}}}\hspace{+0.060em}' - {\vec{\mathit{R}}}\hspace{+0.060em}') ~\doteq~ {\mathbf{a}}\hspace{+0.060em}' ~=~ {\mathbf{a}}$

\par \bigskip {\centering\subsection*{Anexo II}}

\par \medskip {\centering\subsubsection*{Sistema de N Partículas}}

\par \bigskip\smallskip \hspace{-1.80em} ${\mathrm{M}} \,\doteq\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i$

\par \bigskip \hspace{-1.80em} ${\mathbf{R}}_{cm} \,\doteq\, {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{r}}_i$

\par \bigskip \hspace{-1.80em} ${\mathbf{V}}_{cm} \,\doteq\, {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{v}}_i$

\par \bigskip \hspace{-1.80em} ${\mathrm{I}} \,\doteq\, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm M^{\scriptscriptstyle -1}} \, [ \, {\mathbf{r}}_i - {\mathbf{r}}_j \, ]^2$

\par \bigskip \hspace{-1.80em} ${\mathbf{P}} \,\doteq\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{v}}_i$

\par \bigskip \hspace{-1.80em} ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{T}}} \,\doteq\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, ({\mathbf{v}}_i)^2$

\par \bigskip \hspace{-1.80em} ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{P}}} \,\doteq\, \med \; {\mathrm{M}} \, {\mathbf{V}}_{cm}{\hphantom{{\mathbf{V}}}}^{\hspace{-1.92em}2} {\hspace{+0.75em}}\,=\, \med \; {\mathrm{M}}^{\scriptscriptstyle -1} \, {\mathbf{P}}^{\hspace{+0.060em}2}$

\par \bigskip \hspace{-1.80em} ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{A}}} \,\doteq\, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, m_j \, {\mathrm M^{\scriptscriptstyle -1}} \, [ \, ({\mathbf{r}}_i - {\mathbf{r}}_j) \times ({\mathbf{v}}_i - {\mathbf{v}}_j) \hspace{+0.120em}/\hspace{+0.120em} |\hspace{+0.090em}{\mathbf{r}}_i - {\mathbf{r}}_j\hspace{+0.045em}| \: ]^2$

\par \bigskip \hspace{-1.80em} ${\mathrm{K}}^{\scriptscriptstyle{\mathrm{R}}} \,\doteq\, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, m_j \, {\mathrm M^{\scriptscriptstyle -1}} \, [ \, ({\mathbf{r}}_i - {\mathbf{r}}_j) \cdot ({\mathbf{v}}_i - {\mathbf{v}}_j) \hspace{+0.120em}/\hspace{+0.120em} |\hspace{+0.090em}{\mathbf{r}}_i - {\mathbf{r}}_j\hspace{+0.045em}| \: ]^2$

\vspace{+1.50em}

\par \medskip {\centering\subsubsection*{Par de Partículas $i$ y $j$}}

\par \bigskip\smallskip \hspace{-1.80em} $m_i \,\doteq\, m_j \,\doteq\, \med \; {\mathrm{M}} \; \rightarrow \; \mec \: {\mathrm{M}} \,=\, m_i \, m_j \, {\mathrm M^{\scriptscriptstyle -1}}$

\par \bigskip \hspace{-1.80em} ${\mathbf{R}}_{cm} \,\doteq\, {\mathrm{M}}^{\scriptscriptstyle -1} \, ( \, m_i \, {\mathbf{r}}_i + m_j \, {\mathbf{r}}_j \, )$ [ Si ${\mathbf{R}}_{cm} = 0 \; \rightarrow \; {\mathbf{r}}_i = - \, {\mathbf{r}}_j$ ]

\par \bigskip \hspace{-1.80em} ${\mathbf{V}}_{cm} \,\doteq\, {\mathrm{M}}^{\scriptscriptstyle -1} \, ( \, m_i \, {\mathbf{v}}_i + m_j \, {\mathbf{v}}_j \, )$ [ Si ${\mathbf{V}}_{cm} = 0 \; \rightarrow \; {\mathbf{v}}_i = - \, {\mathbf{v}}_j$ ]

\par \bigskip \hspace{-1.80em} ${\mathrm{I}} \,\doteq\, \mec \: {\mathrm{M}} \, [ \, {\mathbf{r}}_i - {\mathbf{r}}_j \, ]^2 \,=\, \mec \: {\mathrm{M}} \: {\mathbf{r}}_{\hspace{+0.060em}ij}^{\hspace{+0.090em}2}$

\par \bigskip \hspace{-1.80em} ${\mathbf{P}} \,\doteq\, m_i \, {\mathbf{v}}_i + m_j \, {\mathbf{v}}_j \,=\, \med \; {\mathrm{M}} \, ( \, {\mathbf{v}}_i + {\mathbf{v}}_j \, )$

\par \bigskip \hspace{-1.80em} ${\mathbf{L}} \,\doteq\, \mec \: {\mathrm{M}} \, [ \, ({\mathbf{r}}_i - {\mathbf{r}}_j) \times ({\mathbf{v}}_i - {\mathbf{v}}_j) \, ] \,=\, {\mathrm{I}} \: {\boldsymbol{\omega}}_{\hspace{+0.060em}ij} \,=\, \mec \: {\mathrm{M}} \: {\boldsymbol{\omega}}_{\hspace{+0.060em}ij} \: {\mathbf{r}}_{\hspace{+0.060em}ij}^{\hspace{+0.090em}2}$

\par \bigskip \hspace{-1.80em} ${\mathrm{G}} \,\doteq\, \mec \: {\mathrm{M}} \, [ \, ({\mathbf{r}}_i - {\mathbf{r}}_j) \cdot ({\mathbf{v}}_i - {\mathbf{v}}_j) \, ] \,=\, {\mathrm{I}} \: {\dot{\mathit{r}}}_{\hspace{+0.060em}ij} \, |\hspace{+0.075em}{\mathbf{r}}_{\hspace{+0.060em}ij}\hspace{+0.075em}|^{\scriptscriptstyle -1} \,=\, \mec \: {\mathrm{M}} \: {\dot{\mathit{r}}}_{\hspace{+0.060em}ij} \, |\hspace{+0.075em}{\mathbf{r}}_{\hspace{+0.060em}ij}\hspace{+0.075em}|$
