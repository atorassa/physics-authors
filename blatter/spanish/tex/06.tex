\par \bigskip {\centering\subsection*{Introducción}}

\par \bigskip \noindent La posición inercial ${\mathbf{r}}_i$, la velocidad inercial ${\mathbf{v}}_i$ y la aceleración inercial ${\mathbf{a}}_{\hspace{+0.045em}i}$ de una partícula $i$, están dadas por:

\par \bigskip \hspace{+0.75em} ${\mathbf{r}}_i \,\doteq\, ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip \hspace{+0.75em} ${\mathbf{v}}_i \,\doteq\, ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip \hspace{+0.75em} ${\mathbf{a}}_{\hspace{+0.045em}i} \,\doteq\, ({\vec{\mathit{a}}}_i - {\vec{\mathit{A}}}) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}}) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \bigskip \noindent {\small $( \, {\mathbf{v}}_i \doteq d(\hspace{+0.090em}{\mathbf{r}}_i\hspace{+0.090em})/dt \, )$} y {\small $( \, {\mathbf{a}}_{\hspace{+0.045em}i} \doteq d^2(\hspace{+0.090em}{\mathbf{r}}_i\hspace{+0.090em})/dt^2 \, )$} donde ${\vec{\mathit{r}}}_i$ es el vector de posición de la partícula $i$, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del free-system y ${\vec{\omega}}$ es el vector de velocidad angular del free-system (ver Anexo I\hspace{+0.120em})

\par \bigskip \noindent La fuerza neta ${\mathbf{F}}_i$ que actúa sobre una partícula $i$ ($m_i$) produce una aceleración \hbox {inercial ${\mathbf{a}}_{\hspace{+0.045em}i}$}, según la siguiente ecuación:

\par \bigskip \hspace{+0.75em} ${\mathbf{F}}_i \,=\, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent Los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias sobre ${\mathbf{F}}_i$.

\par \bigskip \noindent Las magnitudes $[ \, {\mathit{m}}_i$, ${\mathbf{r}}_i$, ${\mathbf{v}}_i$, ${\mathbf{a}}_{\hspace{+0.045em}i}$ y ${\mathbf{F}}_i \, ]$ son invariantes bajo transformaciones entre \hbox {sistemas} de referencia inerciales y no inerciales.

\par \bigskip \noindent Un sistema de referencia S es no rotante si la velocidad angular ${\vec{\omega}}$ del free-system respecto a S es igual a cero y además S es inercial si la aceleración ${\vec{\mathit{A}}}$ del centro de masa del free-system respecto a S es igual a cero.

\par \bigskip {\centering\subsection*{Bipartícula}}

\par \bigskip \noindent Un par de partículas es una bipartícula. El sistema de partículas $\{${\hspace{+0.12em}\it a, b, c\hspace{+0.18em}} y {\it d} $\}$ puede formar el sistema de bipartículas $\{${\hspace{+0.12em}\it ab, ac, ad, bc, bd\hspace{+0.18em}} y {\it cd} $\}$ o también el sistema de bipartículas $\{${\hspace{+0.12em}\it ad, bd\hspace{+0.18em}} y {\it cd} $\}$

\par \bigskip \noindent La masa $m_{\hspace{+0.060em}ij}$ de una bipartícula $ij$, está dada por: $m_{\hspace{+0.060em}ij} \,\doteq\, m_i \, m_j / M$, donde $m_i$ es la masa de las partícula $i$, $m_j$ es la masa de la partícula $j$\hspace{+0.045em} y $M$ {\small $(\, \doteq \sum_k \, m_k)$} es la masa del sistema de partículas en observación.

\par \bigskip \noindent La posición inercial ${\mathbf{r}}_{\hspace{+0.060em}ij}$, la velocidad inercial ${\mathbf{v}}_{\hspace{+0.060em}ij}$ y la aceleración inercial ${\mathbf{a}}_{\hspace{+0.060em}ij}$ de una bipartícula $ij$, están dadas por:

\par \bigskip \hspace{+0.75em} ${\mathbf{r}}_{\hspace{+0.060em}ij} \,\doteq\, ({\vec{\mathit{r}}}_i - {\vec{\mathit{r}}}_j)$

\par \bigskip \hspace{+0.75em} ${\mathbf{v}}_{\hspace{+0.060em}ij} \,\doteq\, ({\vec{\mathit{v}}}_i - {\vec{\mathit{v}}}_j) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{r}}}_j)$

\par \bigskip \hspace{+0.75em} ${\mathbf{a}}_{\hspace{+0.060em}ij} \,\doteq\, ({\vec{\mathit{a}}}_i - {\vec{\mathit{a}}}_j) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_i - {\vec{\mathit{v}}}_j) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{r}}}_j) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{r}}}_j)$

\par \bigskip \noindent {\small $( \, {\mathbf{v}}_{\hspace{+0.060em}ij} \doteq d(\hspace{+0.090em}{\mathbf{r}}_{\hspace{+0.060em}ij}\hspace{+0.090em})/dt \, )$} y {\small $( \, {\mathbf{a}}_{\hspace{+0.060em}ij} \doteq d^2(\hspace{+0.090em}{\mathbf{r}}_{\hspace{+0.060em}ij}\hspace{+0.090em})/dt^2 \, )$} donde ${\vec{\mathit{r}}}_i$ es el vector de posición de la \hbox {partícula $i$}, ${\vec{\mathit{r}}}_j$ es el vector de posición de la partícula $j$ y ${\vec{\omega}}$ es el vector de velocidad angular del free-system (ver Anexo II\hspace{+0.120em})

\par \bigskip \noindent La fuerza neta ${\mathbf{F}}_{\hspace{+0.060em}ij}$ que actúa sobre una bipartícula $ij$ ($m_{\hspace{+0.060em}ij}$) produce una aceleración \hbox {inercial ${\mathbf{a}}_{\hspace{+0.060em}ij}$}, según la siguiente ecuación:

\par \bigskip \hspace{+0.75em} ${\mathbf{F}}_{\hspace{+0.060em}ij} \,\doteq\, m_{\hspace{+0.060em}ij} \, (\hspace{+0.075em}{\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j\hspace{+0.045em}) \,=\, m_{\hspace{+0.060em}ij} \, (\hspace{+0.075em}{\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{a}}_j\hspace{+0.045em}) \,=\, m_{\hspace{+0.060em}ij} \, {\mathbf{a}}_{\hspace{+0.060em}ij}$

\par \bigskip \noindent donde ${\mathbf{F}}_i$ es la fuerza neta que actúa sobre la partícula $i$, ${\mathbf{F}}_j$ es la fuerza neta que actúa sobre la partícula $j$, $m_i$ es la masa de las partícula $i$, $m_j$ es la masa de la \hbox {partícula $j$}, \hbox {${\mathbf{a}}_{\hspace{+0.045em}i}$ es} la aceleración inercial de la partícula $i$ y ${\mathbf{a}}_j$ es la aceleración inercial de la \hbox {partícula $j$}.

\par \bigskip \noindent Los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias sobre ${\mathbf{F}}_i$ ni sobre ${\mathbf{F}}_j$.

\par \bigskip \noindent Las magnitudes $[ \, {\mathit{m}}_{\hspace{+0.060em}ij}$, ${\mathbf{r}}_{\hspace{+0.060em}ij}$, ${\mathbf{v}}_{\hspace{+0.060em}ij}$, ${\mathbf{a}}_{\hspace{+0.060em}ij}$ y ${\mathbf{F}}_{\hspace{+0.060em}ij} \, ]$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \bigskip \noindent Si la partícula $j$ de una bipartícula $ij$ es el centro de masa del free-system (${\mathbf{F}}_j=0$ y ${\mathbf{a}}_j=0$) entonces de la última ecuación de arriba se obtiene:

\par \bigskip \hspace{+0.75em} ${\mathbf{F}}_i \,=\, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent Por lo tanto, se puede considerar que la dinámica de la partícula es un caso especial de la dinámica de la bipartícula.

\par \bigskip \noindent ${\mathbf{F}}_{\scriptscriptstyle{\mathrm{L1}}} ~\doteq~ d({\mathbf{P}}_{\scriptscriptstyle{\mathrm{L1}}})/dt ~=~ m \; d({\mathbf{v}})/dt ~=~ m \, {\mathbf{a}} ~=~ {\mathbf{F}}$

\par \bigskip {\centering\subsection*{Observaciones}}

\par \bigskip\smallskip \noindent Todas las ecuaciones de este trabajo pueden ser aplicadas en cualquier sistema de referencia inercial o no inercial sin necesidad de introducir las fuerzas ficticias sobre ${\mathbf{F}}_i$ ni sobre ${\mathbf{F}}_j$.

\par \bigskip \noindent En este trabajo, las magnitudes $[ \, {\mathit{m}}_i, {\mathbf{r}}_i, {\mathbf{v}}_i, {\mathbf{a}}_{\hspace{+0.045em}i}, {\mathbf{F}}_i, {\mathbf{P}}_i, {\mathrm{F}}_i, {\mathrm{P}}_i, {\mathrm{K}}_i, {\mathrm{U}}_i, {\mathrm{E}}_i, {\mathrm{L}}_i, {\mathit{m}}_{\hspace{+0.060em}ij}, {\mathbf{r}}_{\hspace{+0.060em}ij}, {\mathbf{v}}_{\hspace{+0.060em}ij}$, ${\mathbf{a}}_{\hspace{+0.060em}ij}, {\mathbf{F}}_{\hspace{+0.060em}ij}, {\mathbf{P}}_{\hspace{+0.060em}ij}, {\mathrm{F}}_{\hspace{+0.060em}ij}, {\mathrm{P}}_{\hspace{+0.060em}ij}, {\mathrm{K}}_{\hspace{+0.060em}ij}, {\mathrm{U}}_{\hspace{+0.060em}ij}, {\mathrm{E}}_{\hspace{+0.060em}ij}$ y ${\mathrm{L}}_{\hspace{+0.060em}ij} \, ]$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \bigskip \noindent Este trabajo no contradice la dinámica de Newton. De hecho, la ecuación $[ \, {\mathbf{F}}_i = m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} \, ]$ y la ecuación $[ \, {\mathbf{F}}_{\hspace{+0.060em}ij} = m_{\hspace{+0.060em}ij} \, {\mathbf{a}}_{\hspace{+0.060em}ij} \, ]$ son simples reformulaciones de la segunda ley de Newton.

\par \bigskip \noindent Por último, las integrales usadas en este trabajo son integrales indefinidas. Si ninguna fuerza actúa sobre la partícula $i$, sobre la partícula $j$ o sobre la bipartícula $ij$ entonces las integrales dan como resultado constantes.

\par \bigskip {\centering\subsection*{Anexo I}}

\par \medskip {\centering\subsubsection*{Free-System}}

\par \bigskip \noindent El free-system es un sistema de N partículas que está siempre libre de fuerzas externas e internas, que es tridimensional y que las distancias relativas entre las N partículas permanecen siempre constantes.

\par \bigskip \noindent La posición ${\vec{\mathit{R}}}$, la velocidad ${\vec{\mathit{V}}}$ y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del free-system respecto a un sistema de referencia S, la velocidad angular ${\vec{\omega}}$ y la aceleración \hbox {angular ${\vec{\alpha}}$} del free-system respecto al sistema de referencia S, están dadas por:

\par \bigskip\smallskip \hspace{-2.40em} \begin{tabular}{l}
${\mathrm{M}} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i$ \vspace{+1.20em} \\
${\vec{\mathit{R}}} ~\doteq~ {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\vec{\mathit{r}}_{i}}$ \vspace{+1.20em} \\
${\vec{\mathit{V}}} ~\doteq~ {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\vec{\mathit{v}}_{i}}$ \vspace{+1.20em} \\
${\vec{\mathit{A}}} ~\doteq~ {\mathrm{M}}^{\scriptscriptstyle -1} \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\vec{\mathit{a}}_{i}}$ \vspace{+1.20em} \\
${\vec{\omega}} ~\doteq~ {\mathit{I}}^{\scriptscriptstyle -1}{\vphantom{\sum_1^2}}^{\hspace{-1.500em}\leftrightarrow}\hspace{+0.600em} \cdot {\vec{\mathit{L}}}$ \vspace{+1.20em} \\
${\vec{\alpha}} ~\doteq~ d({\vec{\omega}})/dt$ \vspace{+1.20em} \\
${\mathit{I}}{\vphantom{\sum_1^2}}^{\hspace{-0.555em}\leftrightarrow}\hspace{-0.210em} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, [ \, |\hspace{+0.090em}{\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}\,|^2 \hspace{+0.309em} {\mathrm{1}}{\vphantom{\sum_1^2}}^{\hspace{-0.639em}\leftrightarrow}\hspace{-0.129em} - ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}) \otimes ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}) \, ]$ \vspace{+1.20em} \\
${\vec{\mathit{L}}} ~\doteq~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{R}}}) \times ({\vec{\mathit{v}}_{i}} - \hspace{-0.120em}{\vec{\mathit{V}}})$

\par \bigskip {\centering\subsection*{Anexo III}}

\par \medskip {\centering\subsubsection*{Fuerzas Cinéticas}}

\par \bigskip \noindent La fuerza cinética ${\mathbf{K}}^{a}_{\hspace{+0.060em}ij}$ ejercida sobre una partícula $i$ de masa $m_i$ por otra partícula $j$ de masa $m_j$, causada por la interacción entre la partícula $i$ y la partícula $j$, \hbox {está dada por}:

\par \bigskip \noindent ${\mathbf{K}}^{a}_{\hspace{+0.060em}ij} \,=\, - \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \, (\hspace{+0.045em}{\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{a}}_{j})$

\par \bigskip \noindent donde ${\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración inercial de la partícula $i$, ${\mathbf{a}}_{j}$ es la aceleración inercial de la \hbox {partícula $j$} y ${\mathrm{M}}$ es la masa del Universo.

\par \bigskip \noindent La fuerza cinética ${\mathbf{K}}^{u}_{i}$ ejercida sobre una partícula $i$ de masa $m_i$ por el centro de masa del Universo, causada por la interacción entre la partícula $i$ y el Universo, \hbox {está dada por}:

\par \bigskip \noindent ${\mathbf{K}}^{u}_{i} \,=\, - \; m_i \, {\mathbf{A}}_{cm}$

\par \bigskip \noindent donde ${\mathbf{A}}_{cm}$ es la aceleración inercial del centro de masa del Universo.

\par \bigskip \noindent De las ecuaciones anteriores se deduce que la fuerza cinética neta ${\mathbf{K}}_{i}$ \hbox {{\small (\hspace{+0.120em}$ = \sum_{j}^{\scriptscriptstyle{\mathit{All}}} \, {\mathbf{K}}^{a}_{\hspace{+0.060em}ij} + {\mathbf{K}}^{u}_{i}$\hspace{+0.120em})}} que actúa sobre una partícula $i$ de masa $m_i$, está dada por:

\par \bigskip \noindent ${\mathbf{K}}_{i} \,=\, - \; m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent donde ${\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración inercial de la partícula $i$.

\par \bigskip \noindent Las fuerzas cinéticas ${\mathbf{K}}^{a}$ obedecen siempre la tercera ley de Newton en su forma débil.

\par \bigskip \noindent Si todas las fuerzas no cinéticas obedecen siempre la tercera ley de Newton en su forma débil entonces la aceleración inercial del centro de masa del Universo ${\mathbf{A}}_{cm}$ es siempre igual a cero.

\par \bigskip \noindent Ahora, de la página $[\,1\,]$ y de la página $[\,2\,]$, se tiene:

\par \bigskip \noindent ${\mathbf{F}}_i \,=\, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent ${\mathbf{F}}_{\hspace{+0.060em}ij} \,=\, m_{\hspace{+0.060em}ij} \, {\mathbf{a}}_{\hspace{+0.060em}ij}$

\par \bigskip \noindent O sea:

\par \bigskip \noindent ${\mathbf{F}}_i - m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} \,=\, 0 \,=\, {\mathbf{F}}_i + {\mathbf{K}}_i$

\par \bigskip \noindent ${\mathbf{F}}_{\hspace{+0.060em}ij} - m_{\hspace{+0.060em}ij} \, {\mathbf{a}}_{\hspace{+0.060em}ij} \,=\, 0 \,=\, {\mathbf{F}}_{\hspace{+0.060em}ij} - m_{\hspace{+0.060em}ij} \, (\hspace{+0.075em}{\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{a}}_j\hspace{+0.045em}) \,=\, {\mathbf{F}}_{\hspace{+0.060em}ij} +\, m_{\hspace{+0.060em}ij} \, (\hspace{+0.075em}{\mathbf{K}}_i / m_i - {\mathbf{K}}_j / m_j\hspace{+0.045em}) \,=\, {\mathbf{F}}_{\hspace{+0.060em}ij} + {\mathbf{K}}_{\hspace{+0.060em}ij}$

\par \bigskip \noindent Por lo tanto, la fuerza total $( {\mathbf{F}}_i + {\mathbf{K}}_i )$ que actúa sobre una partícula $i$ y la fuerza total $( {\mathbf{F}}_{\hspace{+0.060em}ij} + {\mathbf{K}}_{\hspace{+0.060em}ij} )$ que actúa sobre una bipartícula $ij$ están siempre en equilibrio.
