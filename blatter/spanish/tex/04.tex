\par \bigskip {\centering\subsection*{Introducción}}

\par \bigskip Un par de partículas es una bipartícula. El sistema de partículas $\{${\hspace{+0.12em}\it a, b, c\hspace{+0.18em}} y {\it d} $\}$ puede formar el sistema de bipartículas $\{${\hspace{+0.12em}\it ab, ac, ad, bc, bd\hspace{+0.18em}} y {\it cd} $\}$ o también el sistema de bipartículas $\{${\hspace{+0.12em}\it ad, bd\hspace{+0.18em}} y {\it cd} $\}$

\par \bigskip La masa $m_{\hspace{+0.060em}ij}$ de una bipartícula $ij$, está dada por: $m_{\hspace{+0.060em}ij} \,\doteq\, m_i \, m_j / M$, donde $m_i$ es la masa de las partícula $i$, $m_j$ es la masa de la partícula $j$\hspace{+0.045em} y $M$ {\small $(\, \doteq \sum_k \, m_k)$} es la masa del sistema de partículas en observación.

\par \bigskip La posición radial ${\mathit{r}}_{\hspace{+0.060em}ij}$, la velocidad radial ${\dot{\mathit{r}}}_{\hspace{+0.060em}ij}$ y la aceleración radial ${\ddot{\mathit{r}}}_{\hspace{+0.060em}ij}$ de una bipartícula $ij$, están dadas por:

\par \bigskip ${\mathit{r}}_{\hspace{+0.060em}ij} \,\doteq\, | \, {\mathbf{r}}_i \hspace{-0.24em}-\hspace{-0.24em} {\mathbf{r}}_j \, |$

\par \bigskip ${\dot{\mathit{r}}}_{\hspace{+0.060em}ij} \,\doteq\, [ \, ({\mathbf{v}}_i \hspace{-0.24em}-\hspace{-0.24em} {\mathbf{v}}_j) \hspace{-0.219em}\cdot\hspace{-0.219em} ({\mathbf{r}}_i \hspace{-0.24em}-\hspace{-0.24em} {\mathbf{r}}_j) \, ] \,/\, | \, {\mathbf{r}}_i \hspace{-0.24em}-\hspace{-0.24em} {\mathbf{r}}_j \, |$

\par \bigskip ${\ddot{\mathit{r}}}_{\hspace{+0.060em}ij} \,\doteq\, [ \, ({\mathbf{a}}_i - {\mathbf{a}}_j) \cdot ({\mathbf{r}}_i - {\mathbf{r}}_j) + ({\mathbf{v}}_i - {\mathbf{v}}_j) \cdot ({\mathbf{v}}_i - {\mathbf{v}}_j) - [ ({\mathbf{v}}_i - {\mathbf{v}}_j) \cdot ({\mathbf{r}}_i - {\mathbf{r}}_j) ]^2 \hspace{-0.06em} / ({\mathbf{r}}_i - {\mathbf{r}}_j)^2 \, ] \,/\, | \, {\mathbf{r}}_i - {\mathbf{r}}_j \, |$

\par \bigskip \noindent donde ${\mathbf{r}}_i$ es el vector de posición de la partícula $i$ y ${\mathbf{r}}_j$ es el vector de posición de la partícula $j$ {\small $( \, {\dot{\mathit{r}}_{\hspace{+0.060em}ij} \doteq d({\mathit{\hspace{+0.045em}r_{\hspace{+0.060em}ij}}})/dt} \, )$} y {\small $( \, {\ddot{\mathit{r}}_{\hspace{+0.060em}ij} \doteq d^2({\mathit{\hspace{+0.045em}r_{\hspace{+0.060em}ij}}})/dt^2} \, )$}

\par \bigskip El momento radial ${\mathit{P}}_{\hspace{+0.060em}ij}$ de una bipartícula $ij$ ($m_{\hspace{+0.060em}ij}$) está dado por: ${\mathit{P}}_{\hspace{+0.060em}ij} \,\doteq\, m_{\hspace{+0.060em}ij} \, {\dot{\mathit{r}}}_{\hspace{+0.060em}ij}$, donde ${\dot{\mathit{r}}}_{\hspace{+0.060em}ij}$ es la velocidad radial de la bipartícula $ij$.

\par \bigskip La fuerza radial ${\mathit{F}}_{\hspace{+0.060em}ij}$ que actúa sobre una bipartícula $ij$ ($m_{\hspace{+0.060em}ij}$) está dada por:

\par \bigskip ${\mathit{F}}_{\hspace{+0.060em}ij} \;\; \,\doteq\, \;\; [ \; m_{\hspace{+0.060em}ij} \,/\: | \, {\mathbf{r}}_i - {\mathbf{r}}_j \, | \; ] \;\; [ \;\; (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot ({\mathbf{r}}_i - {\mathbf{r}}_j) \; + \; 2 \int (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot d({\mathbf{r}}_i - {\mathbf{r}}_j) \;\; -$

\par \bigskip $[ \, 2 \int \, [ \, (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot ({\mathbf{r}}_i - {\mathbf{r}}_j) + 2 \int (\frac{{\mathbf{F}}_i}{m_i} - \frac{{\mathbf{F}}_j}{m_j}) \cdot d({\mathbf{r}}_i - {\mathbf{r}}_j) \, ] \; d\,\med\,({\mathbf{r}}_i - {\mathbf{r}}_j)^2 \, ] \,/\, \hspace{-0.09em} ({\mathbf{r}}_i - {\mathbf{r}}_j)^2 \; ]$

\par \bigskip \noindent donde ${\mathbf{F}}_i$ es la fuerza neta que actúa sobre la partícula $i$, ${\mathbf{F}}_j$ es la fuerza neta que actúa sobre la partícula $j$, $m_i$ es la masa de las partícula $i$, $m_j$ es la masa de la partícula $j$, ${\mathbf{r}}_i$ es el vector de posición de la partícula $i$ y ${\mathbf{r}}_j$ es el vector de posición de la \hbox {partícula $j$.}

\newpage

\par \bigskip {\centering\subsection*{Dinámica Radial}}

\par \bigskip La ecuación radial de movimiento para una bipartícula $ij$ ($m_{\hspace{+0.060em}ij}$) está dada por:

\par \bigskip \hspace{+1.23em} ${\mathit{F}}_{\hspace{+0.060em}ij} \,=\, d({\mathit{P}}_{\hspace{+0.060em}ij})/dt \,=\, m_{\hspace{+0.060em}ij} \, {\ddot{\mathit{r}}}_{\hspace{+0.060em}ij}$

\par \bigskip \noindent donde ${\mathit{F}}_{\hspace{+0.060em}ij}$ es la fuerza radial que actúa sobre la bipartícula $ij$, ${\mathit{P}}_{\hspace{+0.060em}ij}$ es el momento radial de la bipartícula $ij$ y ${\ddot{\mathit{r}}}_{\hspace{+0.060em}ij}$ es la aceleración radial de la bipartícula $ij$.

\par \bigskip\medskip {\centering\subsubsection*{W, K, U, E y L}}

\par \bigskip El trabajo ${\mathit{W}}_{\hspace{+0.060em}ij}$ realizado por las fuerzas (vectoriales) que actúan sobre una bipartícula $ij$, está dado por:

\par \bigskip \hspace{+1.23em} ${\mathit{W}}_{\hspace{+0.060em}ij} \,\doteq\, \int_1^2 \hspace{+0.045em} {\mathit{F}}_{\hspace{+0.060em}ij} \; d\hspace{+0.090em}({\mathit{r}}_{\hspace{+0.060em}ij}) \,=\, \Delta \, \med \; m_{\hspace{+0.060em}ij} \, ({\dot{\mathit{r}}}_{\hspace{+0.060em}ij})^2 \,\doteq\, \Delta \, {\mathit{K}}_{\hspace{+0.060em}ij}$

\par \bigskip \noindent donde ${\mathit{F}}_{\hspace{+0.060em}ij}$ es la fuerza radial que actúa sobre la bipartícula $ij$, ${\mathit{r}}_{\hspace{+0.060em}ij}$ es la posición radial de la bipartícula $ij$, $m_{\hspace{+0.060em}ij}$ es la masa de la bipartícula $ij$, ${\dot{\mathit{r}}}_{\hspace{+0.060em}ij}$ es la velocidad radial de la bipartícula $ij$ y ${\mathit{K}}_{\hspace{+0.060em}ij}$ es la energía cinética de la bipartícula $ij$.

\par \bigskip El trabajo ${\mathit{W}}_{\hspace{+0.060em}ij}$ realizado por las fuerzas (vectoriales) conservativas que actúan sobre una bipartícula $ij$ es igual y de signo opuesto a la variación en la energía potencial ${\mathit{U}}_{\hspace{+0.060em}ij}$ de la bipartícula $ij$.

\par \bigskip \hspace{+1.23em} $\Delta \, {\mathit{U}}_{\hspace{+0.060em}ij} \,\doteq\, - \, \int_1^2 \hspace{+0.045em} {\mathit{F}}_{\hspace{+0.060em}ij} \; d\hspace{+0.090em}({\mathit{r}}_{\hspace{+0.060em}ij})$

\par \bigskip Por lo tanto, la energía mecánica ${\mathit{E}}_{\hspace{+0.060em}ij}$ de una bipartícula $ij$ permanece constante si la bipartícula $ij$ está sujeta sólo a fuerzas (vectoriales) conservativas. 

\par \bigskip \hspace{+1.23em} $\Delta \, {\mathit{E}}_{\hspace{+0.060em}ij} \,\doteq\, \Delta \, {\mathit{K}}_{\hspace{+0.060em}ij} + \Delta \, {\mathit{U}}_{\hspace{+0.060em}ij} \,=\, 0$

\par \bigskip \hspace{+1.23em} ${\mathit{E}}_{\hspace{+0.060em}ij} \,\doteq\, {\mathit{K}}_{\hspace{+0.060em}ij} + {\mathit{U}}_{\hspace{+0.060em}ij} \,=\, {\mathit{constante}}$

\par \bigskip \noindent donde ${\mathit{K}}_{\hspace{+0.060em}ij}$ es la energía cinética de la bipartícula $ij$ y ${\mathit{U}}_{\hspace{+0.060em}ij}$ es la energía potencial de la bipartícula $ij$.

\par \bigskip Finalmente, el Lagrangiano ${\mathit{L}}_{\hspace{+0.060em}ij}$ de una bipartícula $ij$, está dado por:

\par \bigskip \hspace{+1.23em} ${\mathit{L}}_{\hspace{+0.060em}ij} \,\doteq\, {\mathit{K}}_{\hspace{+0.060em}ij} - {\mathit{U}}_{\hspace{+0.060em}ij}$

\par \bigskip \noindent donde ${\mathit{K}}_{\hspace{+0.060em}ij}$ es la energía cinética de la bipartícula $ij$ y ${\mathit{U}}_{\hspace{+0.060em}ij}$ es la energía potencial de la bipartícula $ij$.

\par \bigskip {\centering\subsection*{Observaciones}}

\par \bigskip\smallskip Todas las ecuaciones de este trabajo pueden ser aplicadas en cualquier sistema de referencia inercial o no inercial.

\par \bigskip Los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias sobre ${\mathbf{F}}_i$ ni sobre ${\mathbf{F}}_j$.

\par \bigskip Todas las magnitudes de este trabajo $({\hspace{+0.045em}}{\mathit{r}}_{\hspace{+0.060em}ij}, {\dot{\mathit{r}}}_{\hspace{+0.060em}ij}, {\ddot{\mathit{r}}}_{\hspace{+0.060em}ij}, {\mathit{P}}_{\hspace{+0.060em}ij}, {\mathit{F}}_{\hspace{+0.060em}ij}, {\mathit{W}}_{\hspace{+0.060em}ij}, {\mathit{K}}_{\hspace{+0.060em}ij}, {\mathit{U}}_{\hspace{+0.060em}ij}, {\mathit{E}}_{\hspace{+0.060em}ij}$ y ${\mathit{L}}_{\hspace{+0.060em}ij})$ son invariantes bajo transformaciones entre sistemas de referencia.

\par \bigskip Las magnitudes $({\mathit{W}}_{\hspace{+0.060em}ij}, {\mathit{K}}_{\hspace{+0.060em}ij}, {\mathit{U}}_{\hspace{+0.060em}ij}, {\mathit{E}}_{\hspace{+0.060em}ij}$ y ${\mathit{L}}_{\hspace{+0.060em}ij})$ son en realidad magnitudes escalares \hbox {nuevas que} por comodidad en este trabajo se les quitó el adjetivo <<\hspace{+0.045em}radial\hspace{+0.045em}>>

\par \bigskip Las integrales de la definición de ${\mathit{F}}_{\hspace{+0.060em}ij}$ son integrales indefinidas. Si ninguna fuerza actúa sobre las partículas $i$ y $j$ entonces las integrales dan como resultado constantes.

\par \bigskip La definición de ${\mathit{F}}_{\hspace{+0.060em}ij}$ podría modificarse de manera tal que no haya necesidad de trabajar con integrales indefinidas. Sin embargo, este cambio podría obligar a tener que modificar también otras ecuaciones del trabajo.

\par \bigskip Por otro lado, este trabajo no contradice la dinámica de Newton. De hecho, si un sistema de referencia inercial está fijo sobre la partícula $j$ $({\mathbf{r}}_j={\mathbf{v}}_j={\mathbf{a}}_j={\mathbf{F}}_j=0)$ de una bipartícula $ij$ ($m_{\hspace{+0.060em}ij}$) entonces de la ecuación radial de movimiento, se obtiene:

\par \bigskip \noindent {\small $\frac{{\mathbf{F}}_i}{m_i} \cdot {\mathbf{r}}_i + 2 \int \frac{{\mathbf{F}}_i}{m_i} \cdot d{\mathbf{r}}_i - [ \, 2 \int \, [ \, \frac{{\mathbf{F}}_i}{m_i} \cdot {\mathbf{r}}_i + 2 \int \frac{{\mathbf{F}}_i}{m_i} \cdot d{\mathbf{r}}_i \, ] \; d\,\med\,({\mathbf{r}}_i)^2 \, ] \,/\, \hspace{-0.09em} ({\mathbf{r}}_i)^2 \;=\; {\mathbf{a}}_i \cdot {\mathbf{r}}_i + {\mathbf{v}}_i \cdot {\mathbf{v}}_i - [ \, {\mathbf{v}}_i \cdot {\mathbf{r}}_i \, ]^2 \hspace{-0.06em} / ({\mathbf{r}}_i)^2$}

\par \bigskip \noindent {\small $(\frac{{\mathbf{F}}_i}{m_i} - {\mathbf{a}}_i) \cdot {\mathbf{r}}_i + 2 \int \frac{{\mathbf{F}}_i}{m_i} \cdot d{\mathbf{r}}_i - {\mathbf{v}}_i \cdot {\mathbf{v}}_i - [ \; 2 \int \, [ \, \frac{{\mathbf{F}}_i}{m_i} \cdot {\mathbf{r}}_i + 2 \int \frac{{\mathbf{F}}_i}{m_i} \cdot d{\mathbf{r}}_i \, ] \; d\,\med\,({\mathbf{r}}_i)^2 \, + [ \, {\mathbf{v}}_i \cdot {\mathbf{r}}_i \, ]^2 \; ] \,/\, \hspace{-0.09em} ({\mathbf{r}}_i)^2 \,=\, 0$}

\par \bigskip \noindent {\small $(\frac{{\mathbf{F}}_i}{m_i} - {\mathbf{a}}_i) \cdot {\mathbf{r}}_i = 0 \rightarrow 2 \int \frac{{\mathbf{F}}_i}{m_i} \cdot d{\mathbf{r}}_i - {\mathbf{v}}_i \cdot {\mathbf{v}}_i = 0 \rightarrow 2 \int \, [ \, \frac{{\mathbf{F}}_i}{m_i} \cdot {\mathbf{r}}_i + 2 \int \frac{{\mathbf{F}}_i}{m_i} \cdot d{\mathbf{r}}_i \, ] \; d\,\med\,({\mathbf{r}}_i)^2 - [ \, {\mathbf{v}}_i \cdot {\mathbf{r}}_i \, ]^2 = 0$}

\par \bigskip \noindent puesto que en todo sistema de referencia inercial siempre ${\mathbf{a}}_i = {\mathbf{F}}_i/m_i$ (así como en todo sistema de referencia no inercial introduciendo las fuerzas ficticias sobre ${\mathbf{F}}_i$)

\par \bigskip Finalmente, este trabajo considera que es posible desarrollar una dinámica radial alternativa basada en el Lagrangiano ${\mathit{L}}_{\hspace{+0.060em}ij}$ que puede ser aplicada en cualquier sistema de referencia inercial o no inercial sin necesidad de introducir las fuerzas ficticias.
