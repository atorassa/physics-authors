\par \medskip {\centering\subsection*{Introducción}}\addcontentsline{toc}{subsection}{1. Introducción}

\par \medskip \noindent La nueva reformulación, que este trabajo presenta en mecánica clásica, se desarrolla a partir de un sistema auxiliar de partículas (denominado Universo) que es utilizado para obtener magnitudes cinemáticas (denominadas universales) que son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \medskip \noindent La posición universal ${\mathbf{r}}_i$, velocidad universal ${\mathbf{v}}_i$ y aceleración universal ${\mathbf{a}}_{\hspace{+0.045em}i}$ de una partícula $i$ con respecto a un sistema de referencia S (\hspace{+0.120em}inercial o no inercial\hspace{+0.120em}) están dadas por:

\par \medskip\vspace{+0.06em} ${\mathbf{r}}_i \,\:\doteq\; (\hspace{+0.090em}{\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}\hspace{+0.090em}) \;=\; ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \medskip\vspace{+0.36em} ${\mathbf{v}}_i \;\doteq\; d\hspace{+0.090em}(\hspace{+0.090em}{\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}\hspace{+0.090em})\hspace{+0.045em}/dt \;=\; ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \medskip\vspace{+0.36em} ${\mathbf{a}}_{\hspace{+0.045em}i} \;\doteq\; d^2\hspace{+0.030em}(\hspace{+0.090em}{\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}\hspace{+0.090em})\hspace{+0.045em}/dt^2 \:=\; ({\vec{\mathit{a}}}_i - {\vec{\mathit{A}}}) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}}) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \medskip\vspace{+0.36em} \noindent donde ${\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}$ es el vector de posición de la partícula $i$ con respecto al sistema universal [\hspace{+0.120em}${\vec{\mathit{r}}}_i$ es el vector de posición de la partícula $i$, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del Universo y ${\vec{\omega}}$ es el vector de velocidad angular del Universo\hspace{+0.120em}] [\hspace{+0.120em}con respecto al sistema S\hspace{+0.120em}] \hyperlink{t1a1}{\hbox {(ver {\small A}nexo {\small I}\hspace{+0.120em})}}

\par \medskip\vspace{+0.00em} \noindent El sistema universal es un sistema de referencia fijo al Universo (\hspace{+0.120em}${\vec{\omega}} = 0$\hspace{+0.120em}) cuyo origen coincide siempre con el centro de masa del Universo (\hspace{+0.033em}{\small ${\vec{\mathit{R}}} = {\vec{\mathit{V}}} = {\vec{\mathit{A}}} =$} $0$\hspace{+0.120em})

\par \medskip \noindent Cualquier sistema de referencia S es inercial cuando la velocidad angular ${\vec{\omega}}$ del Universo y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo son iguales a cero \hbox {con respecto a S.}

\par \medskip \noindent Nota\hspace{+0.300em}:\hspace{+0.240em}\hbox{( $\forall \;\; {\mathbf{m}} \, \in$\hspace{+0.12em} Magnitudes Universales \hspace{+0.18em}:\hspace{+0.18em} Si \hspace{+0.24em}${\mathbf{m}} \,=\, {\vec{\mathit{n}}}$ \hspace{+0.27em}$\longrightarrow$\hspace{+0.27em} $d(\hspace{+0.090em}{\mathbf{m}}\hspace{+0.090em})/dt \,=\, d(\hspace{+0.090em}{\vec{\mathit{n}}}\hspace{+0.090em})/dt \,-\, {\vec{\omega}} \times {\vec{\mathit{n}}}$ )}

\vspace{+0.90em}

\par {\centering\subsection*{Nueva Dinámica}}\addcontentsline{toc}{subsection}{2. Nueva Dinámica}

\par \medskip \noindent $[\,1\,]$ Toda fuerza siempre es causada por la interacción entre dos o más partículas.

\par \medskip \noindent $[\,2\,]$ La fuerza neta ${\mathbf{F}}_i$ que actúa sobre una partícula $i$ de masa $m_i$ produce una aceleración universal ${\mathbf{a}}_{\hspace{+0.045em}i}$ según la siguiente ecuación: $[ \, {\mathbf{F}}_i \,=\, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} \, ]$

\par \medskip \noindent $[\,3\,]$ En este trabajo, todas las fuerzas deben obedecer siempre la tercera ley de Newton en su forma débil y en su forma fuerte.

\newpage

\par \bigskip {\centering\subsection*{Ecuación de Movimiento}}\addcontentsline{toc}{subsection}{3. Ecuación de Movimiento}

\par \bigskip \noindent La fuerza neta ${\mathbf{F}}_i$ que actúa sobre una partícula $i$ de masa $m_i$ produce una aceleración universal ${\mathbf{a}}_{\hspace{+0.045em}i}$ según la siguiente ecuación:

\par \bigskip\smallskip ${\mathbf{F}}_i \,=\, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip\smallskip \noindent A partir de la ecuación anterior se deduce que la aceleración (\hspace{+0.120em}ordinaria\hspace{+0.120em}) ${\vec{\mathit{a}}}_{i}$ de la partícula $i$ con respecto a un sistema de referencia S (\hspace{+0.120em}inercial o no inercial\hspace{+0.120em}) está dada por:

\par \bigskip\smallskip ${\vec{\mathit{a}}}_{i} \;=\: {\mathbf{F}}_{\hspace{-0.09em}i}/m_i + {\vec{\mathit{A}}} \hspace{+0.15em} + 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}}) - {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \, ] + {\vec{\alpha}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}})$

\par \bigskip\smallskip \noindent donde ${\vec{\mathit{r}}}_i$ es el vector de posición de la partícula $i$, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del Universo y ${\vec{\omega}}$ es el vector de velocidad angular del Universo \hyperlink{t1a1}{(ver {\small A}nexo {\small I}\hspace{+0.120em})}

\par \bigskip \noindent Desde la ecuación anterior se deduce que la partícula $i$ puede estar acelerada incluso si sobre la partícula $i$ no actúa fuerza alguna y también que la partícula $i$ puede no estar acelerada (\hspace{+0.120em}estado de reposo o de movimiento rectilíneo uniforme\hspace{+0.120em}) incluso si sobre la \hbox {partícula $i$} actúa una fuerza neta no equilibrada.

\par \bigskip \noindent Sin embargo, desde la ecuación anterior también se deduce que la primera y segunda ley de Newton son válidas en cualquier sistema de referencia inercial, puesto que la velocidad angular ${\vec{\omega}}$ del Universo y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo son iguales a cero con respecto a cualquier sistema de referencia inercial.

\par \bigskip \noindent En este trabajo, cualquier sistema de referencia S es un sistema de referencia inercial cuando la velocidad angular ${\vec{\omega}}$ del Universo y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo son iguales a cero con respecto a S. Por lo tanto, cualquier sistema de referencia S es un sistema de referencia no inercial cuando la velocidad angular ${\vec{\omega}}$ del Universo y{\small /}o la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo no son iguales a cero con respecto a S.

\par \bigskip \noindent Sin embargo, puesto que en mecánica clásica cualquier sistema de referencia es realmente un cuerpo rígido ideal entonces cualquier sistema de referencia S es un sistema inercial cuando la fuerza neta que actúa en cada punto del sistema S es igual a cero. Por lo tanto, cualquier sistema de referencia S es un sistema no inercial cuando la fuerza neta que actúa en cada punto del sistema S no es igual a cero \hyperlink{t1a4}{(ver {\small A}nexo {\small IV}\hspace{+0.120em})}

\par \bigskip \noindent Por otro lado, la nueva reformulación de la mecánica clásica presentada en este trabajo y la mecánica newtoniana son observacionalmente equivalentes.

\par \bigskip \noindent Sin embargo, los observadores no inerciales pueden utilizar la mecánica newtoniana sólo si introducen fuerzas ficticias en ${\mathbf{F}}_{\hspace{-0.09em}i}$ (\hspace{+0.120em}como la fuerza centrífuga, la fuerza de \hbox {Coriolis, etc.\hspace{+0.120em})}

\par \bigskip \noindent Adicionalmente, la nueva reformulación de la mecánica clásica presentada en este trabajo es también una reformulación relacional de la mecánica clásica puesto que está desarrollada a partir de magnitudes relativas (\hspace{+0.120em}posición, velocidad y aceleración\hspace{+0.120em}) entre partículas.

\par \bigskip \noindent Sin embargo, como se ha dicho más arriba, la nueva reformulación de la mecánica clásica presentada en este trabajo y la mecánica newtoniana son observacionalmente equivalentes.

\par {\centering\subsection*{Leyes de Conservación}}\addcontentsline{toc}{subsection}{6. Leyes de Conservación}

\par \bigskip\smallskip \noindent El momento lineal $[ \, {\mathbf{P}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas internas obedecen la tercera ley de Newton en su forma débil.

\par \bigskip\medskip ${\mathbf{P}}_1 ~=~ {\mathrm{constante}} \hspace{+2.88em} \big [ \; d({\mathbf{P}}_1)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent El momento angular $[ \, {\mathbf{L}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas internas obedecen la tercera ley de Newton en su forma fuerte.

\par \bigskip\medskip ${\mathbf{L}}_1 ~=~ {\mathrm{constante}} \hspace{+2.97em} \big [ \; d({\mathbf{L}}_1)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, {\mathbf{r}}_i \times {\mathbf{a}}_{\hspace{+0.045em}i} \, \big ]~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, {\mathbf{r}}_i \times {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent El momento angular $[ \, {\mathbf{L}}_2 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas internas obedecen la tercera ley de Newton en su forma fuerte.

\par \bigskip\medskip ${\mathbf{L}}_2 ~=~ {\mathrm{constante}} \hspace{+2.97em} \big [ \; d({\mathbf{L}}_2)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times ({\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{A}}_{cm}) \, \big ] ~=~$

\par \bigskip $\hspace{+10.44em} \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times {\mathbf{a}}_{\hspace{+0.045em}i} \, \big ] ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_1$ y ${\mathrm{E}}_2 \, ]$ de un sistema de N partículas permanecen constantes si el sistema está sujeto solamente a fuerzas conservativas.

\par \bigskip\medskip ${\mathrm{E}}_1 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_1 ~=~ \Delta \; {\mathrm{K}}_1 + \Delta \; {\mathrm{U}}_1 ~=~ 0 \; \big ]$

\par \bigskip ${\mathrm{E}}_2 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_2 ~=~ \Delta \; {\mathrm{K}}_2 + \Delta \; {\mathrm{U}}_2 ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_3$ y ${\mathrm{E}}_4 \, ]$ de un sistema de N partículas son siempre iguales a cero, por lo tanto, permanecen siempre constantes.

\par \bigskip\medskip ${\mathrm{E}}_3 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; {\mathrm{E}}_3 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; \big [ \, m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot {\mathbf{r}}_{i} - {\mathbf{F}}_i \cdot {\mathbf{r}}_{i} \, \big ] ~=~ 0 \; \big ]$

\par \bigskip $\hspace{+10.44em} \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, \big [ \, ({\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{A}}_{cm}) \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm}) \, \big ] \,=\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm})$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_5$ y ${\mathrm{E}}_6 \, ]$ de un sistema de N partículas permanecen constantes si el sistema está sujeto solamente a fuerzas conservativas.

\par \bigskip {\centering\subsection*{Observaciones Generales}}\addcontentsline{toc}{subsection}{7. Observaciones Generales}

\par \bigskip \noindent Todas las ecuaciones de este trabajo pueden ser aplicadas en cualquier sistema de referencia inercial o no inercial.

\par \bigskip \noindent Por lo tanto, la nueva reformulación de la mecánica clásica presentada en este trabajo está de acuerdo con el principio general de relatividad.

\par \bigskip \noindent Adicionalmente, los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias en ${\mathbf{F}}_i$ ( como la fuerza centrífuga, la fuerza de Coriolis, etc. )

\par \bigskip \noindent En este trabajo, las magnitudes $[ \, {\mathit{m}},\, {\mathbf{r}},\, {\mathbf{v}},\, {\mathbf{a}},\, {\mathrm{M}},\, {\mathbf{R}}, {\mathbf{V}}, {\mathbf{A}},\, {\mathbf{F}},\, {\mathbf{P}}_1, {\mathbf{L}}_1, {\mathbf{L}}_2, {\mathrm{W}}_1, {\mathrm{K}}_1, {\mathrm{U}}_1, {\mathrm{E}}_1, {\mathrm{L}}_1$, ${\mathrm{W}}_2,\spc {\mathrm{K}}_2,\spc {\mathrm{U}}_2,\spc {\mathrm{E}}_2,\spc {\mathrm{L}}_2,\spc {\mathrm{W}}_3,\spc {\mathrm{K}}_3,\spc {\mathrm{U}}_3,\spc {\mathrm{E}}_3,\spc {\mathrm{W}}_4,\spc {\mathrm{K}}_4,\spc {\mathrm{U}}_4,\spc {\mathrm{E}}_4,\spc {\mathrm{W}}_5,\spc {\mathrm{K}}_5,\spc {\mathrm{U}}_5,\spc {\mathrm{E}}_5,\spc {\mathrm{W}}_6,\spc {\mathrm{K}}_6,\spc {\mathrm{U}}_6$ y ${\mathrm{E}}_6 \, ]$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no \hbox {inerciales}.

\par \bigskip \noindent La energía mecánica ${\mathrm{E}}_3$ de un sistema de partículas es siempre igual a cero $[ \, {\mathrm{E}}_3 = {\mathrm{K}}_3 + {\mathrm{U}}_3 = 0 \, ]$

\par \bigskip \noindent Por lo tanto, la energía mecánica ${\mathrm{E}}_5$ de un sistema de partículas es siempre igual a la energía mecánica ${\mathrm{E}}_1$ del sistema de partículas $[ \, {\mathrm{E}}_5 = {\mathrm{E}}_1 \, ]$

\par \bigskip \noindent La energía mecánica ${\mathrm{E}}_4$ de un sistema de partículas es siempre igual a cero $[ \, {\mathrm{E}}_4 = {\mathrm{K}}_4 + {\mathrm{U}}_4 = 0 \, ]$

\par \bigskip \noindent Por lo tanto, la energía mecánica ${\mathrm{E}}_6$ de un sistema de partículas es siempre igual a la energía mecánica ${\mathrm{E}}_2$ del sistema de partículas $[ \, {\mathrm{E}}_6 = {\mathrm{E}}_2 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_1$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} entonces la energía potencial ${\mathrm{U}}_3$ y la energía potencial ${\mathrm{U}}_5$ del sistema de partículas, están dadas por: $[ \, {\mathrm{U}}_3 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 \, ]$ y $[ \, {\mathrm{U}}_5 = ({\scriptstyle 1 +} \frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_2$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} entonces la energía potencial ${\mathrm{U}}_4$ y la energía potencial ${\mathrm{U}}_6$ del sistema de partículas, están dadas por: $[ \, {\mathrm{U}}_4 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 \, ]$ y $[ \, {\mathrm{U}}_6 = ({\scriptstyle 1 +} \frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_1$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} y si la energía cinética ${\mathrm{K}}_5$ del sistema de partículas es igual a cero entonces se obtiene: $[ \, {\mathrm{K}}_1 = - \, {\mathrm{K}}_3 = {\mathrm{U}}_3 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 = (\frac{{\mathit{k}}}{2 + {\mathit{k}}}) \, {\mathrm{E}}_1 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_2$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} y si la energía cinética ${\mathrm{K}}_6$ del sistema de partículas es igual a cero entonces se obtiene: $[ \, {\mathrm{K}}_2 = - \, {\mathrm{K}}_4 = {\mathrm{U}}_4 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 = (\frac{{\mathit{k}}}{2 + {\mathit{k}}}) \, {\mathrm{E}}_2 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_1$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} y si el promedio de la energía cinética $\langle {\mathrm{K}}_5 \rangle$ del sistema de partículas es igual a cero entonces se obtiene: $[ \, \langle {\mathrm{K}}_1 \rangle = - \, \langle {\mathrm{K}}_3 \rangle = \langle {\mathrm{U}}_3 \rangle = (\frac{{\mathit{k}}}{2}) \, \langle {\mathrm{U}}_1 \rangle = (\frac{{\mathit{k}}}{2 + {\mathit{k}}}) \, \langle {\mathrm{E}}_1 \rangle \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_2$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} y si el promedio de la energía cinética $\langle {\mathrm{K}}_6 \rangle$ del sistema de partículas es igual a cero entonces se obtiene: $[ \, \langle {\mathrm{K}}_2 \rangle = - \, \langle {\mathrm{K}}_4 \rangle = \langle {\mathrm{U}}_4 \rangle = (\frac{{\mathit{k}}}{2}) \, \langle {\mathrm{U}}_2 \rangle = (\frac{{\mathit{k}}}{2 + {\mathit{k}}}) \, \langle {\mathrm{E}}_2 \rangle \, ]$

\par \bigskip \noindent El promedio de la energía cinética $\langle {\mathrm{K}}_5 \rangle$ y el promedio de la energía cinética $\langle {\mathrm{K}}_6 \rangle$ de un sistema de partículas con desplazamiento acotado están relacionados con el teorema del virial.

\newpage

\par \bigskip \noindent El promedio de la energía cinética $\langle {\mathrm{K}}_5 \rangle$ y el promedio de la energía cinética $\langle {\mathrm{K}}_6 \rangle$ de un sistema de partículas con desplazamiento acotado ( en $\langle {\mathrm{K}}_5 \rangle$ con respecto a ${\vec{\mathit{R}}}$ \hspace{+0.090em}y\hspace{+0.090em} en $\langle {\mathrm{K}}_6 \rangle$ con respecto a ${\vec{\mathit{R}}}_{cm}$ ) son siempre iguales a cero.

\par \bigskip \noindent La energía cinética ${\mathrm{K}}_5$ y la energía cinética ${\mathrm{K}}_6$ de un sistema de N partículas pueden ser también expresadas como sigue: $[ \; {\mathrm{K}}_5 = \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, ( \, {\dot{\mathit{r}}}_{i} \, {\dot{\mathit{r}}}_{i} + {\ddot{\mathit{r}}}_{i} \, {\mathit{r}}_{i} \, ) \; ]$ donde ${\mathit{r}}_{i} \doteq | \, {\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}} \, |$ \hbox {y $[ \; {\mathrm{K}}_6 = \sum_{i{\scriptscriptstyle <}j}^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} ( \, {\dot{\mathit{r}}}_{\hspace{+0.060em}ij} \, {\dot{\mathit{r}}}_{\hspace{+0.060em}ij} + {\ddot{\mathit{r}}}_{\hspace{+0.060em}ij} \, {\mathit{r}}_{\hspace{+0.060em}ij} \, ) \; ]$} donde ${\mathit{r}}_{\hspace{+0.060em}ij} \,\doteq\, | \: {\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j} \: |$ \hfill {\tiny N1\hspace{-0.03em}} $\scriptstyle \big ( \sum_{i{\scriptscriptstyle <}j}^{\scriptscriptstyle{\mathrm{N}}} \doteq \sum_{i=1}^{\scriptscriptstyle{\mathrm{N}}} \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \big )$

\par \bigskip \noindent La energía cinética ${\mathrm{K}}_5$ y la energía cinética ${\mathrm{K}}_6$ de un sistema de N partículas pueden ser también expresadas como sigue: $[ \; {\mathrm{K}}_5 = \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, ( \, {\ddot{\tau}}_{\hspace{+0.045em}i} \, ) \; ]$ donde ${\tau}_{i} \doteq \med \; ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \cdot ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}})$ \hbox {y $[ \; {\mathrm{K}}_6 = \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} ( \, {\ddot{\tau}}_{\hspace{+0.060em}ij} \, ) \; ]$} donde ${\tau}_{\hspace{+0.060em}ij} \doteq \med \; ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) \cdot ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j})$ \hfill {\tiny N2\hspace{-0.03em}} $\scriptstyle \big ( \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \doteq \sum_{i=1}^{\scriptscriptstyle{\mathrm{N}}} \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \big )$

\par \bigskip \noindent La energía cinética ${\mathrm{K}}_6$ es la única energía cinética que puede ser expresada sin necesidad \hbox {de introducir} magnitud alguna relacionada con el Universo $[ \: $tales como: ${\mathbf{r}},\: {\mathbf{v}},\: {\mathbf{a}},\: {\vec{\omega}},\: {\vec{\mathit{R}}}$, etc.$\: ]$

\par \bigskip \noindent En un sistema aislado de partículas la energía potencial ${\mathrm{U}}_2$ es igual a la energía potencial ${\mathrm{U}}_1$ si las fuerzas internas obedecen la tercera ley de Newton en su forma débil $[ \, {\mathrm{U}}_2 = {\mathrm{U}}_1 \, ]$

\par \bigskip \noindent En un sistema aislado de partículas la energía potencial ${\mathrm{U}}_4$ es igual a la energía potencial ${\mathrm{U}}_3$ si las fuerzas internas obedecen la tercera ley de Newton en su forma débil $[ \, {\mathrm{U}}_4 = {\mathrm{U}}_3 \, ]$

\par \bigskip \noindent En un sistema aislado de partículas la energía potencial ${\mathrm{U}}_6$ es igual a la energía potencial ${\mathrm{U}}_5$ si las fuerzas internas obedecen la tercera ley de Newton en su forma débil $[ \, {\mathrm{U}}_6 = {\mathrm{U}}_5 \, ]$

\par \bigskip \noindent Un sistema de referencia S es un sistema no rotante especial cuando la velocidad angular ${\vec{\omega}}$ del Universo \hbox {con respecto a S} es igual a cero y S es además un sistema inercial cuando la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo con respecto a S es igual a cero.

\par \bigskip \noindent Si el origen de un sistema no rotante especial $[ \, {\vec{\omega}} = 0 \, ]$ coincide siempre con el centro de masa del Universo $[ \, {\vec{\mathit{R}}} = {\vec{\mathit{V}}} = {\vec{\mathit{A}}} = 0 \, ]$ entonces se logra: $[ \, {\mathbf{r}}_i = {\vec{\mathit{r}}}_i$, ${\mathbf{v}}_i = {\vec{\mathit{v}}}_i$ y ${\mathbf{a}}_{\hspace{+0.045em}i} = {\vec{\mathit{a}}}_i \, ]$ \hbox {Por lo tanto}, es posible afirmar que las magnitudes universales y las magnitudes ordinarias son siempre las mismas en este sistema de referencia.

\par \bigskip \noindent Este trabajo no contradice la primera y segunda ley de Newton puesto que estas dos leyes siguen siendo válidas en cualquier sistema de referencia inercial. La ecuación $[ \, {\mathbf{F}}_i = m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} \, ]$ es una simple reformulación de la segunda ley de Newton.

\par \bigskip \noindent Finalmente, en este trabajo, la ecuación $[ \, {\mathbf{F}}_i = m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} \, ]$ es válida en cualquier sistema de referencia inercial o no inercial solamente si todas las fuerzas obedecen siempre la tercera ley de Newton en su forma débil y en su forma fuerte.

\par \medskip {\centering\subsubsection*{El Universo}}\addcontentsline{toc}{subsection}{Anexo I : El Universo}\hypertarget{t1a1}{}

\par \bigskip \noindent El Universo es un sistema que contiene a todas las partículas, que está siempre libre de fuerzas externas y que todas las fuerzas internas obedecen siempre la tercera ley de Newton en su forma débil y en su forma fuerte.

\par \medskip {\centering\subsection*{Introducción}}\addcontentsline{toc}{subsection}{1. Introducción}

\par \medskip \noindent La nueva reformulación, que este trabajo presenta en mecánica clásica, se desarrolla a partir de un sistema auxiliar de partículas (denominado Universo) que es utilizado para obtener magnitudes cinemáticas (denominadas universales) que son invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \medskip \noindent La posición universal ${\mathbf{r}}_i$, velocidad universal ${\mathbf{v}}_i$ y aceleración universal ${\mathbf{a}}_{\hspace{+0.045em}i}$ de una partícula $i$ con respecto a un sistema de referencia S (\hspace{+0.120em}inercial o no inercial\hspace{+0.120em}) están dadas por:

\par \medskip\vspace{+0.06em} ${\mathbf{r}}_i \,\:\doteq\; (\hspace{+0.090em}{\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}\hspace{+0.090em}) \;=\; ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \medskip\vspace{+0.36em} ${\mathbf{v}}_i \;\doteq\; d\hspace{+0.090em}(\hspace{+0.090em}{\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}\hspace{+0.090em})\hspace{+0.045em}/dt \;=\; ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \medskip\vspace{+0.36em} ${\mathbf{a}}_{\hspace{+0.045em}i} \;\doteq\; d^2\hspace{+0.030em}(\hspace{+0.090em}{\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}\hspace{+0.090em})\hspace{+0.045em}/dt^2 \:=\; ({\vec{\mathit{a}}}_i - {\vec{\mathit{A}}}) - 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) + {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}}) \, ] - {\vec{\alpha}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}})$

\par \medskip\vspace{+0.36em} \noindent donde ${\stackrel{\scriptscriptstyle\sim}{\smash{r}\rule{0pt}{+0.30em}}}_{\hspace{-0.12em}i}$ es el vector de posición de la partícula $i$ con respecto al sistema universal [\hspace{+0.120em}${\vec{\mathit{r}}}_i$ es el vector de posición de la partícula $i$, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del Universo y ${\vec{\omega}}$ es el vector de velocidad angular del Universo\hspace{+0.120em}] [\hspace{+0.120em}con respecto al sistema S\hspace{+0.120em}] \hyperlink{t2a1}{\hbox {(ver {\small A}nexo {\small I}\hspace{+0.120em})}}

\par \medskip\vspace{+0.00em} \noindent El sistema universal es un sistema de referencia fijo al Universo (\hspace{+0.120em}${\vec{\omega}} = 0$\hspace{+0.120em}) cuyo origen coincide siempre con el centro de masa del Universo (\hspace{+0.033em}{\small ${\vec{\mathit{R}}} = {\vec{\mathit{V}}} = {\vec{\mathit{A}}} =$} $0$\hspace{+0.120em})

\par \medskip \noindent Cualquier sistema de referencia S es inercial cuando la velocidad angular ${\vec{\omega}}$ del Universo y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo son iguales a cero \hbox {con respecto a S.}

\par \medskip \noindent Nota\hspace{+0.300em}:\hspace{+0.240em}\hbox{( $\forall \;\; {\mathbf{m}} \, \in$\hspace{+0.12em} Magnitudes Universales \hspace{+0.18em}:\hspace{+0.18em} Si \hspace{+0.24em}${\mathbf{m}} \,=\, {\vec{\mathit{n}}}$ \hspace{+0.27em}$\longrightarrow$\hspace{+0.27em} $d(\hspace{+0.090em}{\mathbf{m}}\hspace{+0.090em})/dt \,=\, d(\hspace{+0.090em}{\vec{\mathit{n}}}\hspace{+0.090em})/dt \,-\, {\vec{\omega}} \times {\vec{\mathit{n}}}$ )}

\vspace{+0.90em}

\par {\centering\subsection*{Nueva Dinámica}}\addcontentsline{toc}{subsection}{2. Nueva Dinámica}

\par \medskip \noindent $[\,1\,]$ Toda fuerza siempre es causada por la interacción entre dos o más partículas.

\par \medskip \noindent $[\,2\,]$ La fuerza total ${\mathbf{T}}_i$ que actúa sobre una partícula $i$ es siempre igual a cero $[ \, {\mathbf{T}}_i \,=\, 0 \, ]$

\par \medskip \noindent $[\,3\,]$ En este trabajo, todas las fuerzas dinámicas ( todas las fuerzas no cinéticas ) deben obedecer siempre la tercera ley \hbox {de Newton} en su forma débil y en su forma fuerte.

\newpage

\par \bigskip {\centering\subsection*{Fuerza Cinética}}\addcontentsline{toc}{subsection}{3. Fuerza Cinética}

\par \bigskip \noindent La fuerza cinética ${\mathbf{K}}_{\hspace{+0.060em}ij}$ ejercida sobre una partícula $i$ de masa $m_i$ por otra partícula $j$ de masa $m_j$, causada por la interacción entre la partícula $i$ y la partícula $j$, está dada por:

\par \bigskip ${\mathbf{K}}_{\hspace{+0.060em}ij} ~=~ - \; \dfrac{m_i \, m_j}{\mathit{M}} \, (\hspace{+0.045em}{\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{a}}_{j})$

\par \bigskip \noindent donde ${\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración universal de la partícula $i$, ${\mathbf{a}}_{j}$ es la aceleración universal de la \hbox {partícula $j$} y ${\mathit{M}}$ {\small \hspace{-0.120em}(\hspace{+0.120em}$ = \sum_i^{\scriptscriptstyle{\mathit{All}}} \, m_i$\hspace{+0.120em})} es la masa del Universo.

\par \bigskip \noindent De la ecuación anterior se deduce que la fuerza cinética neta ${\mathbf{K}}_{i}$ {\small \hbox {(\hspace{+0.120em}$ = \sum_{j}^{\scriptscriptstyle{\mathit{All}}} \, {\mathbf{K}}_{\hspace{+0.060em}ij}$\hspace{+0.120em})}} que actúa sobre una partícula $i$ de masa $m_i$, está dada por:

\par \bigskip ${\mathbf{K}}_{i} ~=~ - \; m_i \, (\hspace{+0.045em}{\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{A}})$

\par \bigskip \noindent donde ${\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración universal de la partícula $i$ y ${\mathbf{A}}$ {\small \hspace{-0.120em}(\hspace{+0.120em}$ = {\mathit{M}}^{\scriptscriptstyle -1} \sum_i^{\scriptscriptstyle{\mathit{All}}} \, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$\hspace{+0.120em})} es la aceleración universal del centro de masa del Universo.

\par \bigskip \noindent Puesto que la aceleración universal del centro de masa del Universo ${\mathbf{A}}$ es siempre igual a cero entonces la fuerza cinética \hbox {neta ${\mathbf{K}}_{i}$} que actúa sobre una partícula $i$ de masa $m_i$ \hspace{+0.030em}queda\hspace{+0.090em}:

\par \bigskip ${\mathbf{K}}_{i} ~=~ - \; m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent donde ${\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración universal de la partícula $i$.

\par \bigskip \noindent La fuerza cinética ${\mathbf{K}}$ es considerada en la nueva dinámica, principalmente en el $[\,2\,]$ principio, como una nueva fuerza universal de interacción.

\par \bigskip \noindent Por otro lado, la fuerza cinética ${\mathbf{K}}$ obedece siempre la tercera ley de Newton en su forma débil y{\small /}o en su forma fuerte.

\vspace{+0.66em}

\par {\centering\subsection*{[$\,$2$\,$] Principio}}\addcontentsline{toc}{subsection}{4. Segundo Principio}

\par \bigskip \noindent El $[\,2\,]$ principio de la nueva dinámica establece que la fuerza total ${\mathbf{T}}_i$ que actúa sobre una partícula $i$ es siempre igual a cero.

\par \bigskip ${\mathbf{T}}_i ~=~ 0$

\par \bigskip \noindent Si la fuerza total ${\mathbf{T}}_i$ es dividida en las siguientes dos partes: la fuerza cinética neta ${\mathbf{K}}_{i}$ y la fuerza dinámica neta ${\mathbf{F}}_{i}$ (\hspace{+0.060em}$\sum$ de fuerzas gravitatorias, fuerzas electrostáticas, etc.\hspace{+0.060em}) entonces:

\par \bigskip ${\mathbf{K}}_{i} + {\mathbf{F}}_{i} ~=~ 0$

\par \bigskip \noindent Ahora, sustituyendo ${\mathbf{K}}_{i}$ {\normalsize \hspace{-0.120em}(\hspace{+0.120em}$ = - \; m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$\hspace{+0.120em})} y reordenando, finalmente se obtiene:

\par \bigskip ${\mathbf{F}}_{i} ~=~ m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip \noindent Esta ecuación (\hspace{+0.060em}similar a la segunda ley de Newton\hspace{+0.060em}) será usada a lo largo de este trabajo.

\par \bigskip \noindent Por otro lado, en este trabajo un sistema de partículas es aislado cuando el sistema está libre de fuerzas dinámicas externas.

\par \bigskip {\centering\subsection*{Ecuación de Movimiento}}\addcontentsline{toc}{subsection}{5. Ecuación de Movimiento}

\par \bigskip \noindent La fuerza dinámica neta ${\mathbf{F}}_i$ que actúa sobre una partícula $i$ de masa $m_i$ está relacionada con la aceleración universal ${\mathbf{a}}_{\hspace{+0.045em}i}$ de la partícula $i$ según la siguiente ecuación:

\par \bigskip\smallskip ${\mathbf{F}}_i \,=\, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\par \bigskip\smallskip \noindent A partir de la ecuación anterior se deduce que la aceleración (\hspace{+0.120em}ordinaria\hspace{+0.120em}) ${\vec{\mathit{a}}}_{i}$ de la partícula $i$ con respecto a un sistema de referencia S (\hspace{+0.120em}inercial o no inercial\hspace{+0.120em}) está dada por:

\par \bigskip\smallskip ${\vec{\mathit{a}}}_{i} \;=\: {\mathbf{F}}_{\hspace{-0.09em}i}/m_i + {\vec{\mathit{A}}} \hspace{+0.15em} + 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}}) - {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \, ] + {\vec{\alpha}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}})$

\par \bigskip\smallskip \noindent donde ${\vec{\mathit{r}}}_i$ es el vector de posición de la partícula $i$, ${\vec{\mathit{R}}}$ es el vector de posición del centro de masa del Universo y ${\vec{\omega}}$ es el vector de velocidad angular del Universo \hyperlink{t2a1}{(ver {\small A}nexo {\small I}\hspace{+0.120em})}

\par \bigskip \noindent Desde la ecuación anterior se deduce que la partícula $i$ puede estar acelerada incluso si sobre la partícula $i$ no actúa fuerza dinámica alguna y también que la partícula $i$ puede no estar acelerada (\hspace{+0.120em}estado de reposo o de movimiento rectilíneo uniforme\hspace{+0.120em}) incluso si sobre la \hbox {partícula $i$} actúa una fuerza dinámica neta no equilibrada.

\par \bigskip \noindent Sin embargo, desde la ecuación anterior también se deduce que la primera y segunda ley de Newton son válidas en cualquier sistema de referencia inercial, puesto que la velocidad angular ${\vec{\omega}}$ del Universo y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo son iguales a cero con respecto a cualquier sistema de referencia inercial.

\par \bigskip \noindent En este trabajo, cualquier sistema de referencia S es un sistema de referencia inercial cuando la velocidad angular ${\vec{\omega}}$ del Universo y la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo son iguales a cero con respecto a S. Por lo tanto, cualquier sistema de referencia S es un sistema de referencia no inercial cuando la velocidad angular ${\vec{\omega}}$ del Universo y{\small /}o la aceleración ${\vec{\mathit{A}}}$ del centro de masa del Universo no son iguales a cero con respecto a S.

\par \bigskip \noindent Sin embargo, puesto que en mecánica clásica cualquier sistema de referencia es realmente un cuerpo rígido ideal entonces cualquier sistema de referencia S es un sistema inercial cuando la fuerza dinámica neta que actúa en cada punto del sistema S es igual a cero. Por lo tanto, cualquier sistema de referencia S es un sistema no inercial cuando la fuerza dinámica neta que actúa en cada punto del sistema S no es igual a cero \hyperlink{t2a4}{(ver {\small A}nexo {\small IV}\hspace{+0.120em})}

\par \bigskip \noindent Por otro lado, la nueva reformulación de la mecánica clásica presentada en este trabajo y la mecánica newtoniana son observacionalmente equivalentes.

\par \bigskip \noindent Sin embargo, los observadores no inerciales pueden utilizar la mecánica newtoniana sólo si introducen fuerzas ficticias en ${\mathbf{F}}_{\hspace{-0.09em}i}$ (\hspace{+0.120em}como la fuerza centrífuga, la fuerza de \hbox {Coriolis, etc.\hspace{+0.120em})}

\par \bigskip \noindent Adicionalmente, la nueva reformulación de la mecánica clásica presentada en este trabajo es también una reformulación relacional de la mecánica clásica puesto que está desarrollada a partir de magnitudes relativas (\hspace{+0.120em}posición, velocidad y aceleración\hspace{+0.120em}) entre partículas.

\par \bigskip \noindent Sin embargo, como se ha dicho más arriba, la nueva reformulación de la mecánica clásica presentada en este trabajo y la mecánica newtoniana son observacionalmente equivalentes.

\par {\centering\subsection*{Leyes de Conservación}}\addcontentsline{toc}{subsection}{8. Leyes de Conservación}

\par \bigskip\smallskip \noindent El momento lineal $[ \, {\mathbf{P}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas dinámicas internas obedecen la tercera ley de Newton en su forma débil.

\par \bigskip\medskip ${\mathbf{P}}_1 ~=~ {\mathrm{constante}} \hspace{+2.88em} \big [ \; d({\mathbf{P}}_1)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, {\mathbf{a}}_{\hspace{+0.045em}i} ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent El momento angular $[ \, {\mathbf{L}}_1 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas dinámicas internas obedecen la tercera ley de Newton en su forma fuerte.

\par \bigskip\medskip ${\mathbf{L}}_1 ~=~ {\mathrm{constante}} \hspace{+2.97em} \big [ \; d({\mathbf{L}}_1)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, {\mathbf{r}}_i \times {\mathbf{a}}_{\hspace{+0.045em}i} \, \big ]~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, {\mathbf{r}}_i \times {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent El momento angular $[ \, {\mathbf{L}}_2 \, ]$ de un sistema aislado de N partículas permanece constante si las fuerzas dinámicas internas obedecen la tercera ley de Newton en su forma fuerte.

\par \bigskip\medskip ${\mathbf{L}}_2 ~=~ {\mathrm{constante}} \hspace{+2.97em} \big [ \; d({\mathbf{L}}_2)/dt ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times ({\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{A}}_{cm}) \, \big ] ~=~$

\par \bigskip $\hspace{+10.44em} \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, \big [ \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times {\mathbf{a}}_{\hspace{+0.045em}i} \, \big ] ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, ({\mathbf{r}}_i - {\mathbf{R}}_{cm}) \times {\mathbf{F}}_i ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_1$ y ${\mathrm{E}}_2 \, ]$ de un sistema de N partículas permanecen constantes si el sistema está sujeto solamente a fuerzas cinéticas y a fuerzas dinámicas conservativas.

\par \bigskip\medskip ${\mathrm{E}}_1 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_1 ~=~ \Delta \; {\mathrm{K}}_1 + \Delta \; {\mathrm{U}}_1 ~=~ 0 \; \big ]$

\par \bigskip ${\mathrm{E}}_2 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_2 ~=~ \Delta \; {\mathrm{K}}_2 + \Delta \; {\mathrm{U}}_2 ~=~ 0 \; \big ]$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_3$ y ${\mathrm{E}}_4 \, ]$ de un sistema de N partículas son siempre iguales a cero, por lo tanto, permanecen siempre constantes.

\par \bigskip\medskip ${\mathrm{E}}_3 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; {\mathrm{E}}_3 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; \big [ \, m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot {\mathbf{r}}_{i} - {\mathbf{F}}_i \cdot {\mathbf{r}}_{i} \, \big ] ~=~ 0 \; \big ]$

\par \bigskip ${\mathrm{E}}_4 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; {\mathrm{E}}_4 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; \big [ \, m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm}) - {\mathbf{F}}_i \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm}) \, \big ] ~=~ 0 \; \big ]$

\par \bigskip $\hspace{+10.44em} \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \, \big [ \, ({\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{A}}_{cm}) \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm}) \, \big ] \,=\, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \, \med \; m_i \: {\mathbf{a}}_{\hspace{+0.045em}i} \cdot ({\mathbf{r}}_{i} - {\mathbf{R}}_{cm})$

\par \bigskip\medskip \noindent Las energías mecánicas $[ \, {\mathrm{E}}_5$ y ${\mathrm{E}}_6 \, ]$ de un sistema de N partículas permanecen constantes si el sistema está sujeto solamente a fuerzas cinéticas y a fuerzas dinámicas conservativas.

\par \bigskip\medskip ${\mathrm{E}}_5 ~=~ {\mathrm{constante}} \hspace{+3.00em} \big [ \; \Delta \; {\mathrm{E}}_5 ~=~ \Delta \; {\mathrm{K}}_5 + \Delta \; {\mathrm{U}}_5 ~=~ 0 \; \big ]$

\par \bigskip {\centering\subsection*{Observaciones Generales}}\addcontentsline{toc}{subsection}{9. Observaciones Generales}

\par \bigskip \noindent Todas las ecuaciones de este trabajo pueden ser aplicadas en cualquier sistema de referencia inercial o no inercial.

\par \bigskip \noindent Por lo tanto, la nueva reformulación de la mecánica clásica presentada en este trabajo está de acuerdo con el principio general de relatividad.

\par \bigskip \noindent Adicionalmente, los sistemas de referencia inerciales y no inerciales no deben introducir las fuerzas ficticias en ${\mathbf{F}}_i$ ( como la fuerza centrífuga, la fuerza de Coriolis, etc. )

\par \bigskip \noindent En este trabajo, las magnitudes $[ \, {\mathit{m}}, {\mathbf{r}}, {\mathbf{v}}, {\mathbf{a}}, {\mathrm{M}}, {\mathbf{R}}, {\mathbf{V}}\hspace{-0.120em}, {\mathbf{A}}, {\mathbf{T}}, {\mathbf{K}}, {\mathbf{F}}, {\mathbf{P}}_1, {\mathbf{L}}_1, {\mathbf{L}}_2, {\mathrm{W}}_1, {\mathrm{K}}_1, {\mathrm{U}}_1, {\mathrm{E}}_1, {\mathrm{L}}_1$ ${\mathrm{W}}_2,\spc {\mathrm{K}}_2,\spc {\mathrm{U}}_2,\spc {\mathrm{E}}_2,\spc {\mathrm{L}}_2,\spc {\mathrm{W}}_3,\spc {\mathrm{K}}_3,\spc {\mathrm{U}}_3,\spc {\mathrm{E}}_3,\spc {\mathrm{W}}_4,\spc {\mathrm{K}}_4,\spc {\mathrm{U}}_4,\spc {\mathrm{E}}_4,\spc {\mathrm{W}}_5,\spc {\mathrm{K}}_5,\spc {\mathrm{U}}_5,\spc {\mathrm{E}}_5,\spc {\mathrm{W}}_6,\spc {\mathrm{K}}_6,\spc {\mathrm{U}}_6$ y ${\mathrm{E}}_6 \, ]$ son invariantes bajo transformaciones entre sistemas de referencia inerciales y no \hbox {inerciales}.

\par \bigskip \noindent La energía mecánica ${\mathrm{E}}_3$ de un sistema de partículas es siempre igual a cero $[ \, {\mathrm{E}}_3 = {\mathrm{K}}_3 + {\mathrm{U}}_3 = 0 \, ]$

\par \bigskip \noindent Por lo tanto, la energía mecánica ${\mathrm{E}}_5$ de un sistema de partículas es siempre igual a la energía mecánica ${\mathrm{E}}_1$ del sistema de partículas $[ \, {\mathrm{E}}_5 = {\mathrm{E}}_1 \, ]$

\par \bigskip \noindent La energía mecánica ${\mathrm{E}}_4$ de un sistema de partículas es siempre igual a cero $[ \, {\mathrm{E}}_4 = {\mathrm{K}}_4 + {\mathrm{U}}_4 = 0 \, ]$

\par \bigskip \noindent Por lo tanto, la energía mecánica ${\mathrm{E}}_6$ de un sistema de partículas es siempre igual a la energía mecánica ${\mathrm{E}}_2$ del sistema de partículas $[ \, {\mathrm{E}}_6 = {\mathrm{E}}_2 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_1$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} entonces la energía potencial ${\mathrm{U}}_3$ y la energía potencial ${\mathrm{U}}_5$ del sistema de partículas, están dadas por: $[ \, {\mathrm{U}}_3 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 \, ]$ y $[ \, {\mathrm{U}}_5 = ({\scriptstyle 1 +} \frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_2$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} entonces la energía potencial ${\mathrm{U}}_4$ y la energía potencial ${\mathrm{U}}_6$ del sistema de partículas, están dadas por: $[ \, {\mathrm{U}}_4 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 \, ]$ y $[ \, {\mathrm{U}}_6 = ({\scriptstyle 1 +} \frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_1$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} y si la energía cinética ${\mathrm{K}}_5$ del sistema de partículas es igual a cero entonces se obtiene: $[ \, {\mathrm{K}}_1 = - \, {\mathrm{K}}_3 = {\mathrm{U}}_3 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_1 = (\frac{{\mathit{k}}}{2 + {\mathit{k}}}) \, {\mathrm{E}}_1 \, ]$

\par \bigskip \noindent Si la energía potencial ${\mathrm{U}}_2$ de un sistema de partículas es una función homogénea de \hbox {grado ${\mathit{k}}$} y si la energía cinética ${\mathrm{K}}_6$ del sistema de partículas es igual a cero entonces se obtiene: $[ \, {\mathrm{K}}_2 = - \, {\mathrm{K}}_4 = {\mathrm{U}}_4 = (\frac{{\mathit{k}}}{2}) \, {\mathrm{U}}_2 = (\frac{{\mathit{k}}}{2 + {\mathit{k}}}) \, {\mathrm{E}}_2 \, ]$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_2 ~=~ - \, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot d({\mathbf{r}}_{i} - {\mathbf{r}}_{j}) \, \big ]$

\par \bigskip\bigskip \noindent ${\mathrm{W}}_4 ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot ({\mathbf{r}}_{i} - {\mathbf{r}}_{j}) \, \big ]$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{K}}_4 ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, ({\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{a}}_{j}) \cdot ({\mathbf{r}}_{i} - {\mathbf{r}}_{j}) \, \big ] ~=~ {\mathrm{W}}_4$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_4 ~=~ - \, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot ({\mathbf{r}}_{i} - {\mathbf{r}}_{j}) \, \big ]$

\par \bigskip\bigskip \noindent ${\mathrm{W}}_6 ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) + \Delta \, \med \; ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ]$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{K}}_6 ~=~ \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, \Delta \, \med \; m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \, ({\vec{\mathit{v}}_{i}} - {\vec{\mathit{v}}_{j}})^2 + ({\vec{\mathit{a}}_{i}} - {\vec{\mathit{a}}_{j}}) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ] ~=~ {\mathrm{W}}_6$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_6 ~=~ - \, \sum_{j{\scriptscriptstyle >}i}^{\scriptscriptstyle{\mathrm{N}}} \, m_i \, m_j \, {\mathrm{M}}^{\scriptscriptstyle -1} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot d({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) + \Delta \, \med \; ({\mathbf{F}}_i / m_i - {\mathbf{F}}_j / m_j) \cdot ({\vec{\mathit{r}}_{i}} - {\vec{\mathit{r}}_{j}}) \, \big ]$

\par \bigskip\bigskip \noindent Las magnitudes ${\mathrm{W}}_{(1\;{\mathrm{al}}\;6)}$ y ${\mathrm{U}}_{(1\;{\mathrm{al}}\;6)}$ de un sistema aislado de N partículas cuyas fuerzas dinámicas internas obedecen la tercera ley de Newton en su forma débil se reducen a:

\par \bigskip\bigskip \noindent ${\mathrm{W}}_1 ~=~ {\mathrm{W}}_2 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}}$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_1 ~=~ \Delta \, {\mathrm{U}}_2 ~=~ - \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}}$

\par \bigskip\bigskip \noindent ${\mathrm{W}}_3 ~=~ {\mathrm{W}}_4 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}}$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_3 ~=~ \Delta \, {\mathrm{U}}_4 ~=~ - \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}}$

\par \bigskip\bigskip \noindent ${\mathrm{W}}_5 ~=~ {\mathrm{W}}_6 ~=~ \sum_i^{\scriptscriptstyle{\mathrm{N}}} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}} + \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}} \, \big ]$

\par \bigskip\smallskip \noindent $\Delta \, {\mathrm{U}}_5 ~=~ \Delta \, {\mathrm{U}}_6 ~=~ - \, \sum_i^{\scriptscriptstyle{\mathrm{N}}} \big [ \int_{\scriptscriptstyle 1}^{\scriptscriptstyle 2} \, {\mathbf{F}}_i \cdot d{\vec{\mathit{r}}_{i}} + \Delta \, \med \; {\mathbf{F}}_i \cdot {\vec{\mathit{r}}_{i}} \, \big ]$

\par \smallskip \noindent La fuerza cinética neta ${\mathbf{K}}_{\hspace{+0.009em}i}$ que actúa sobre una partícula $i$ de masa $m_i$ puede ser también expresada como sigue:

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}i} \;=\: + \; m_i \, \Big[ \: {\mathbf{E}} \, + ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}}) \times {\mathbf{B}} \, \Big]$

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}i} \;=\: + \; m_i \, \Big[ - \nabla \phi \, - \dfrac{\partial{\mathbf{A}}}{\partial{\hspace{+0.03em}t}} + ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}}) \times (\nabla \times {\mathbf{A}}) \, \Big]$

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}i} \;=\: + \; m_i \, \Big[ - ({\vec{\mathit{a}}}_{i} - {\vec{\mathit{A}}}\hspace{+0.06em}) + 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}}) - {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \, ] + {\vec{\alpha}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \, \Big]$

\par \bigskip \noindent donde:

\par \bigskip ${\mathbf{E}} \;=\: - \; \nabla \phi \, - \dfrac{\partial{\mathbf{A}}}{\partial{\hspace{+0.03em}t}}$

\par \bigskip ${\mathbf{B}} \;=\: \nabla \times {\mathbf{A}}$

\par \bigskip $\phi \;=\: - \; \med \: [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \, ]^{\hspace{+0.006em}2} \, + \, \med \, ( {\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}})^{\hspace{+0.03em}2}$

\par \bigskip ${\mathbf{A}} \;=\: - \; [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \, ] \, + ( {\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}})$

\par \bigskip $\dfrac{\partial{\mathbf{A}}}{\partial{\hspace{+0.03em}t}} \;=\: - \; {\vec{\alpha}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) + ( {\vec{\mathit{a}}}_{i} - {\vec{\mathit{A}}}\hspace{+0.06em})$ *

\par \bigskip $\nabla \phi \;=\: {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{R}}}) \, ] \hspace{+0.06em}$

\par \bigskip $\nabla \times {\mathbf{A}} \;=\: - \; 2 \; {\vec{\omega}}$

\par \bigskip \noindent La fuerza cinética neta ${\mathbf{K}}_{\hspace{+0.009em}i}$ que actúa sobre una partícula $i$ de masa $m_i$ puede ser también obtenida a partir de la siguiente energía cinética:

\par \bigskip ${\mathit{K}}_{\hspace{+0.009em}i} \;=\: - \; m_i \, \big[ \, \phi \, - ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{V}}}) \cdot {\mathbf{A}} \, \big]$

\par \bigskip ${\mathit{K}}_{\hspace{+0.009em}i} \;=\: \med \; m_i \, \big[ \, ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{V}}}) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{R}}}) \, \big]^2$

\par \bigskip ${\mathit{K}}_{\hspace{+0.009em}i} \;=\: \med \; m_i \, \big[ \, {\mathbf{v}}_i \, \big]^2$

\vspace{+0.03em}

\par \bigskip \noindent Dado que la energía cinética ${\mathit{K}}_{\hspace{+0.009em}i}$ debe ser positiva, entonces aplicando la siguiente ecuación Euler-Lagrange, se obtiene:

\vspace{+0.06em}

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}i} \;=\: - \; \dfrac{d}{dt} \, \Bigg[ \, \dfrac{\partial{\, \med \; m_i \, \big[ \, {\mathbf{v}}_i \, \big]^2}}{\partial{\,\mathbf{v}}_i} \, \Bigg] + \dfrac{\partial{\, \med \; m_i \, \big[ \, {\mathbf{v}}_i \, \big]^2}}{\partial{\,\mathbf{r}}_i} \;=\: - \; m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\vspace{+0.09em}

\par \bigskip \noindent donde ${\mathbf{r}}_i, {\mathbf{v}}_i$ y ${\mathbf{a}}_{\hspace{+0.045em}i}$ son la posición universal, la velocidad universal y la aceleración universal de la partícula $i$.

\par \bigskip ${\mathbf{B}} \;=\: \nabla \times {\mathbf{A}}$

\par \bigskip $\dfrac{\partial{\mathbf{A}}}{\partial{\hspace{+0.03em}t}} \;=\: - \; {\vec{\alpha}} \times {\vec{\mathit{r}}}_{i} \, + ({\vec{\mathit{a}}}_{i} + {\mathbf{a}}_s)$ *

\par \bigskip $\nabla \phi \;=\: {\vec{\omega}} \times ( \, {\vec{\omega}} \times {\vec{\mathit{r}}}_{i} \, )$

\par \bigskip $\nabla \times {\mathbf{A}} \;=\: - \; 2 \; {\vec{\omega}}$

\par \bigskip \noindent La fuerza cinética neta ${\mathbf{K}}_{\hspace{+0.009em}i}$ que actúa sobre una partícula $i$ de masa $m_i$ puede ser también obtenida a partir de la siguiente energía cinética:

\par \bigskip ${\mathit{K}}_{\hspace{+0.009em}i} \;=\: - \; m_i \, \big[ \, \phi \, - ({\vec{\mathit{v}}}_{i} + {\mathbf{v}}_s) \cdot {\mathbf{A}} \, \big]$

\par \bigskip ${\mathit{K}}_{\hspace{+0.009em}i} \;=\: \med \; m_i \, \big[ \, ({\vec{\mathit{v}}}_{i} + {\mathbf{v}}_s) - ( \, {\vec{\omega}} \times {\vec{\mathit{r}}}_{i} \, ) \, \big]^2$

\vspace{+0.06em}

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}i} \;=\: - \; \dfrac{d}{dt} \, \Bigg[ \, \dfrac{\partial{\, \med \; m_i \, \big[ \, {\mathbf{v}}_i \, \big]^2}}{\partial{\,\mathbf{v}}_i} \, \Bigg] + \dfrac{\partial{\, \med \; m_i \, \big[ \, {\mathbf{v}}_i \, \big]^2}}{\partial{\,\mathbf{r}}_i} \;=\: - \; m_i \, {\mathbf{a}}_{\hspace{+0.045em}i}$

\vspace{+0.09em}

\par {\centering\subsection*{Campos y Potenciales III}}\addcontentsline{toc}{subsection}{Apéndice C : Campos y Potenciales III}

\par \smallskip \noindent La fuerza cinética ${\mathbf{K}}_{\hspace{+0.009em}ij}$ ejercida sobre una partícula $i$ de masa $m_i$ por otra partícula $j$ de masa $m_j$ puede ser también expresada como sigue:

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}ij} \;=\: + \; m_i \: m_j \, {\mathit{M}}^{\scriptscriptstyle -1} \, \Big[ \: {\mathbf{E}} \, + ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{v}}}_{j}) \times {\mathbf{B}} \, \Big]$

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}ij} \;=\: + \; m_i \: m_j \, {\mathit{M}}^{\scriptscriptstyle -1} \, \Big[ - \nabla \phi \, - \dfrac{\partial{\mathbf{A}}}{\partial{\hspace{+0.03em}t}} + ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{v}}}_{j}) \times (\nabla \times {\mathbf{A}}) \, \Big]$

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}ij} \;=\: + \; m_i \: m_j \, {\mathit{M}}^{\scriptscriptstyle -1} \, \Big[ - ({\vec{\mathit{a}}}_{i} - {\vec{\mathit{a}}}_{j}\hspace{+0.06em}) + 2 \; {\vec{\omega}} \times ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{v}}}_{j}) - {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) \, ] + {\vec{\alpha}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) \, \Big]$

\par \bigskip \noindent donde:

\par \bigskip ${\mathbf{E}} \;=\: - \; \nabla \phi \, - \dfrac{\partial{\mathbf{A}}}{\partial{\hspace{+0.03em}t}}$

\par \bigskip ${\mathbf{B}} \;=\: \nabla \times {\mathbf{A}}$

\par \bigskip $\phi \;=\: - \; \med \: [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) \, ]^{\hspace{+0.006em}2} \, + \, \med \, ( {\vec{\mathit{v}}}_{i} - {\vec{\mathit{v}}}_{j})^{\hspace{+0.03em}2}$

\par \bigskip ${\mathbf{A}} \;=\: - \; [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) \, ] \, + ( {\vec{\mathit{v}}}_{i} - {\vec{\mathit{v}}}_{j})$

\par \bigskip $\dfrac{\partial{\mathbf{A}}}{\partial{\hspace{+0.03em}t}} \;=\: - \; {\vec{\alpha}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) + ( {\vec{\mathit{a}}}_{i} - {\vec{\mathit{a}}}_{j}\hspace{+0.06em})$ *

\par \bigskip $\nabla \phi \;=\: {\vec{\omega}} \times [ \, {\vec{\omega}} \times ({\vec{\mathit{r}}}_{i} - {\vec{\mathit{r}}}_{j}) \, ] \hspace{+0.06em}$

\par \bigskip $\nabla \times {\mathbf{A}} \;=\: - \; 2 \; {\vec{\omega}}$

\par \bigskip \noindent La fuerza cinética ${\mathbf{K}}_{\hspace{+0.009em}ij}$ ejercida sobre una partícula $i$ de masa $m_i$ por otra partícula $j$ de masa $m_j$ puede ser también obtenida a partir de la siguiente energía cinética:

\par \bigskip ${\mathit{K}}_{\hspace{+0.03em}ij} \;=\: - \; m_i \: m_j \, {\mathit{M}}^{\scriptscriptstyle -1} \, \big[ \, \phi \, - ({\vec{\mathit{v}}}_{i} - {\vec{\mathit{v}}}_{j}) \cdot {\mathbf{A}} \, \big]$

\par \bigskip ${\mathit{K}}_{\hspace{+0.03em}ij} \;=\: \med \; m_i \: m_j \, {\mathit{M}}^{\scriptscriptstyle -1} \, \big[ \, ({\vec{\mathit{v}}}_i - \hspace{-0.120em}{\vec{\mathit{v}}}_{j}) - {\vec{\omega}} \times ({\vec{\mathit{r}}}_i - {\vec{\mathit{r}}}_{j}) \, \big]^2$

\par \bigskip ${\mathit{K}}_{\hspace{+0.03em}ij} \;=\: \med \; m_i \: m_j \, {\mathit{M}}^{\scriptscriptstyle -1} \, \big[ \, {\mathbf{v}}_i - {\mathbf{v}}_{\hspace{-0.06em}j} \, \big]^2$

\vspace{+0.03em}

\par \bigskip \noindent Dado que la energía cinética ${\mathit{K}}_{\hspace{+0.03em}ij}$ debe ser positiva, entonces aplicando la siguiente ecuación Euler-Lagrange, se obtiene:

\vspace{+0.06em}

\par \bigskip ${\mathbf{K}}_{\hspace{+0.009em}ij} \;=\: - \; \dfrac{d}{dt} \, \Bigg[ \, \dfrac{\partial \, \med \, \frac{m_i \, m_j}{\mathit{M}} \big[ \, {\mathbf{v}}_i - {\mathbf{v}}_{\hspace{-0.06em}j} \, \big]^2}{\partial \, [ \, {\mathbf{v}}_i - {\mathbf{v}}_{\hspace{-0.06em}j} \, ]} \, \Bigg] + \dfrac{\partial \, \med \, \frac{m_i \, m_j}{\mathit{M}} \big[ \, {\mathbf{v}}_i - {\mathbf{v}}_{\hspace{-0.06em}j} \, \big]^2}{\partial \, [ \, {\mathbf{r}}_i - {\mathbf{r}}_{\hspace{-0.015em}j} \, ]} \;=\: - \; \dfrac{m_i \, m_j}{\mathit{M}} \, \big[ \, {\mathbf{a}}_{\hspace{+0.045em}i} - {\mathbf{a}}_{\hspace{-0.015em}j} \, \big]$

\vspace{+0.09em}
