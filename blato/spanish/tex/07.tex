\par \bigskip {\centering\subsubsection*{Introducción}}

\bigskip \smallskip

\noindent En relatividad especial, la posición total $( \, \bar{\mathbf{r}} \, )$ de una partícula (masiva o no masiva) es siempre cero.
\par \vspace{-0.60em}
\begin{eqnarray*}
\bar{\mathbf{r}} ~=~ 0
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent La posición total $( \, \bar{\mathbf{r}} \, )$ de una partícula (masiva o no masiva) está definida por la posición cinética $( \, \hat{\mathbf{r}} \, )$ y la posición dinámica $( \, \check{\mathbf{r}} \, )$ tal como sigue:
\par \vspace{-0.60em}
\begin{eqnarray*}
\hat{\mathbf{r}} - \check{\mathbf{r}} ~=~ 0
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent La posición cinética $( \, \hat{\mathbf{r}} \, )$ de una partícula (masiva o no masiva) está dada por:
\par \vspace{-0.60em}
\begin{eqnarray*}
\hat{\mathbf{r}} ~\doteq~ \dfrac{1}{\mu} \, \int m \, \mathbf{v} \, dt
\end{eqnarray*}
\par \vspace{+0.45em}
\noindent donde $( \, \mu \, )$ es una constante ( universal ) arbitraria, $( \, m \, )$ es la masa relativista de la partícula, $( \, \mathbf{v} \, )$ es la velocidad de la partícula y $( \, t \, )$ es el tiempo.
\par \vspace{+0.60em}
\noindent La posición dinámica $( \, \check{\mathbf{r}} \, )$ de una partícula (masiva o no masiva) está dada por:
\par \vspace{-0.60em}
\begin{eqnarray*}
\check{\mathbf{r}} ~\doteq~ \dfrac{1}{\mu} \, \int \hspace{-0.30em} \int \mathbf{F} \; dt \, dt
\end{eqnarray*}
\par \vspace{+0.45em}
\noindent donde $( \, \mu \, )$ es la constante ( universal ) arbitraria, $( \, \mathbf{F} \, )$ es la fuerza neta que actúa sobre la partícula y $( \, t \, )$ es el tiempo.

\newpage

\noindent La masa relativista $( \, m \, )$ de una partícula masiva está dada por:
\par \vspace{-0.60em}
\begin{eqnarray*}
m ~\doteq~ \dfrac{m_o}{\sqrt{1 - \frac{v^2}{c^2}}}
\end{eqnarray*}
\par \vspace{+0.45em}
\noindent donde $( \, m_o \, )$ es la masa en reposo de la partícula masiva, $( \, v \, )$ es la velocidad de la partícula masiva y $( \, c \, )$ es la velocidad de la luz en el vacío.
\par \vspace{+0.60em}
\noindent La masa relativista $( \, m \, )$ de una partícula no masiva está dada por:
\par \vspace{-0.60em}
\begin{eqnarray*}
m ~\doteq~ \dfrac{h \, \nu}{c^2}
\end{eqnarray*}
\par \vspace{+0.45em}
\noindent donde $( \, h \, )$ es la constante de Planck, $( \, \nu \, )$ es la frecuencia de la partícula no masiva y $( \, c \, )$ es la velocidad de la luz en el vacío.
\par \vspace{+0.75em}
\noindent Ahora, la posición total $( \, \bar{\mathbf{r}} \, )$ de una partícula (masiva o no masiva) puede ser también expresada como sigue:
\par \vspace{-0.60em}
\begin{eqnarray*}
\dfrac{1}{\mu} \; \bigg [ \,\, \int m \, \mathbf{v} \, dt \; - \int \hspace{-0.30em} \int \mathbf{F} \; dt \, dt \,\, \bigg ] =~ 0
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent Derivando la ecuación anterior con respecto al tiempo, resulta:
\par \vspace{-0.60em}
\begin{eqnarray*}
\dfrac{1}{\mu} \; \bigg [ \,\, m \, \mathbf{v} \; - \int \mathbf{F} \; dt \,\, \bigg ] =~ 0
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent Derivando nuevamente con respecto al tiempo, se obtiene:
\par \vspace{-0.60em}
\begin{eqnarray*}
\dfrac{1}{\mu} \; \bigg [ \,\, m \, \mathbf{a} \, + \dfrac{dm}{dt} \, \mathbf{v} \; - \; \mathbf{F} \,\, \bigg ] =~ 0
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent Multiplicando por $( \, \mu \, )$ y reordenando, finalmente se tiene:
\par \vspace{-0.45em}
\begin{eqnarray*}
\mathbf{F} ~=~ m \, \mathbf{a} \, + \dfrac{dm}{dt} \, \mathbf{v}
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent Esta ecuación ( similar a la segunda ley de Newton cuando $v \ll c$ ) será usada en la próxima sección de este artículo.

\par \bigskip {\centering\subsubsection*{Dinámica Relativista}}

\bigskip \smallskip

\noindent Sea una partícula (masiva o no masiva) con masa relativista $m$ entonces el momento lineal $\mathbf{P}$ de la partícula, el momento angular $\mathbf{L}$ de la partícula, la fuerza neta $\mathbf{F}$ que actúa sobre la partícula, el trabajo $\mathrm{W}$ realizado por la fuerza neta que actúa sobre la partícula y la energía cinética $\mathrm{K}$ de la partícula, para un sistema de referencia inercial, están dados por:
\par \vspace{-0.30em}
\begin{eqnarray*}
\mathbf{P} ~\doteq~ m \, \mathbf{v}
\end{eqnarray*}
\vspace{-0.60em}
\begin{eqnarray*}
\mathbf{L} ~\doteq~ \mathbf{P} \times \mathbf{r}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathbf{F} ~=~ \dfrac{d\hspace{+0.045em}\mathbf{P}}{dt} ~=~ m \, \mathbf{a} \, + \dfrac{dm}{dt} \, \mathbf{v}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathrm{W} ~\doteq~ \hspace{-0.36em} \int_{\scriptscriptstyle 1}^{\hspace{+0.09em}{\scriptscriptstyle 2}} \mathbf{F} \cdot d\mathbf{r} ~=~ \hspace{-0.36em} \int_{\scriptscriptstyle 1}^{\hspace{+0.09em}{\scriptscriptstyle 2}} \dfrac{d\hspace{+0.045em}\mathbf{P}}{dt} \cdot d\mathbf{r} ~=~ \Delta \, \mathrm{K}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathrm{K} ~\doteq~ m \, c^2
\end{eqnarray*}
\par \vspace{+0.90em}
\noindent donde $( \, \mathbf{r}, \, \mathbf{v}, \, \mathbf{a} \, )$ son la posición, la velocidad y la aceleración de la partícula respecto al sistema de referencia inercial y $( \, c \, )$ es la velocidad de la luz en el vacío. La energía cinética $( \, \mathrm{K}_o \, )$ de una partícula masiva en reposo es $( \, m_o \, c^2 \, )$
