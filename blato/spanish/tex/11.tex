\par \bigskip {\centering\subsubsection*{Introducción}}

\bigskip \smallskip

\noindent La masa intrínseca $( \, m \, )$ y el factor frecuencia $( \, f \, )$ de una partícula masiva \hbox {están dados por}:
\par \vspace{-0.60em}
\begin{eqnarray*}
m ~\doteq~ m_o
\end{eqnarray*}
\vspace{-0.90em}
\begin{eqnarray*}
f ~\doteq~ \Big ( 1 - \dfrac{\mathbf{v} \cdot \mathbf{v}}{c^2} \hspace{+0.15em} \Big )^{\hspace{-0.24em}-\hspace{+0.03em}1/2}
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent donde $( \, m_o \, )$ es la masa en reposo de la partícula masiva, $( \, \mathbf{v} \, )$ es la velocidad relacional de la partícula masiva y $( \, c \, )$ es la velocidad de la luz en el vacío.
\par \vspace{+0.60em}
\noindent La masa intrínseca $( \, m \, )$ y el factor frecuencia $( \, f \, )$ de una partícula no masiva \hbox {están dados por}:
\par \vspace{-0.60em}
\begin{eqnarray*}
m ~\doteq~ \dfrac{h \, \kappa}{c^2}
\end{eqnarray*}
\vspace{-0.60em}
\begin{eqnarray*}
f ~\doteq~ \dfrac{\nu}{\kappa}
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent donde $( \hspace{+0.33em} h \hspace{+0.33em} )$ es la constante de Planck, \hspace{+0.06em}$( \hspace{+0.30em} \nu \hspace{+0.30em} )$ es la frecuencia relacional de la partícula no masiva, $( \, \kappa \, )$ es una constante universal positiva con dimensión de frecuencia y $( \, c \, )$ es la velocidad de la luz en el vacío.
\par \vspace{+0.60em}
\noindent En este artículo, una partícula masiva es una partícula con masa en reposo no nula y una partícula no masiva es una partícula con masa en reposo nula.

\par \bigskip {\centering\subsubsection*{Cinemática Invariante}}

\bigskip \smallskip

\noindent La posición especial $( \, \bar{\mathbf{r}} \, )$, la velocidad especial $( \, \bar{\mathbf{v}} \, )$ y la aceleración especial $( \, \bar{\mathbf{a}} \, )$ de una partícula ( masiva o no masiva ) están dadas por:
\par \vspace{-0.30em}
\begin{eqnarray*}
\bar{\mathbf{r}} ~\doteq~ \int f \, \mathbf{v} \; d\hspace{+0.012em}t
\end{eqnarray*}
\vspace{-0.45em}
\begin{eqnarray*}
\bar{\mathbf{v}} ~\doteq~ \dfrac{d\hspace{+0.036em}\bar{\mathbf{r}}}{d\hspace{+0.012em}t} ~=~ f \, \mathbf{v}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\bar{\mathbf{a}} ~\doteq~ \dfrac{d\hspace{+0.021em}\bar{\mathbf{v}}}{d\hspace{+0.012em}t} ~=~ f \, \dfrac{d\hspace{+0.021em}\mathbf{v}}{d\hspace{+0.012em}t} + \dfrac{d\hspace{-0.12em}f}{d\hspace{+0.012em}t} \, \mathbf{v}
\end{eqnarray*}
\par \vspace{+1.20em}
\noindent donde $( \, f \, )$ es el factor frecuencia de la partícula, $( \, \mathbf{v} \, )$ es la velocidad relacional de la partícula y $( \, t \, )$ es el tiempo relacional de la partícula.

\vspace{+0.60em}

\par \bigskip {\centering\subsubsection*{Dinámica Invariante}}

\bigskip \smallskip

\noindent Sea una partícula ( masiva o no masiva ) con masa intrínseca $( \, m \, )$ entonces el \hbox {momento} lineal $( \, \mathbf{P} \, )$ de la partícula, el momento angular $( \, \mathbf{L} \, )$ de la partícula \hbox {la fuerza} neta $( \, \mathbf{F} \, )$ que actúa sobre la partícula, el trabajo $( \, \mathrm{W} \, )$ realizado por la fuerza neta que actúa sobre la partícula y la energía cinética $( \, \mathrm{K} \, )$ de la partícula están dados por:
\par \vspace{-0.30em}
\begin{eqnarray*}
\mathbf{P} ~\doteq~ m \, \bar{\mathbf{v}} ~=~ m \, f \, \mathbf{v}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathbf{L} ~\doteq~ \mathbf{P} \hspace{+0.24em}\dot{\times}\hspace{+0.30em} \mathbf{r} ~=~ m \, \bar{\mathbf{v}} \hspace{+0.24em}\dot{\times}\hspace{+0.30em} \mathbf{r} ~=~ m \, f \, \mathbf{v} \hspace{+0.24em}\dot{\times}\hspace{+0.30em} \mathbf{r}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathbf{F} ~=~ \dfrac{d\hspace{+0.045em}\mathbf{P}}{d\hspace{+0.012em}t} ~=~ m \, \bar{\mathbf{a}} ~=~ m \, \bigg [ \, f \, \dfrac{d\hspace{+0.021em}\mathbf{v}}{d\hspace{+0.012em}t} + \dfrac{d\hspace{-0.12em}f}{d\hspace{+0.012em}t} \, \mathbf{v} \, \bigg ]
\end{eqnarray*}
\vspace{-0.15em}
\begin{eqnarray*}
\mathrm{W} ~\doteq~ \hspace{-0.36em} \int_{\scriptscriptstyle 1}^{\hspace{+0.09em}{\scriptscriptstyle 2}} \mathbf{F} \cdot d\hspace{+0.036em}\mathbf{r} ~=~ \hspace{-0.36em} \int_{\scriptscriptstyle 1}^{\hspace{+0.09em}{\scriptscriptstyle 2}} \dfrac{d\hspace{+0.045em}\mathbf{P}}{d\hspace{+0.012em}t} \cdot d\hspace{+0.036em}\mathbf{r} ~=~ \Delta \, \mathrm{K}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathrm{K} ~\doteq~ m \, f \, c^2
\end{eqnarray*}
\par \vspace{+0.90em}
\noindent donde $( \: f, \: \mathbf{r}, \: \mathbf{v}, \: t, \: \bar{\mathbf{v}}, \: \bar{\mathbf{a}} \: )$ son el factor frecuencia, la posición relacional, la velocidad relacional, el tiempo relacional, la velocidad especial y la aceleración especial de la partícula y $( \, c \, )$ es la velocidad de la luz en el vacío. La energía cinética $( \, \mathrm{K}_o \, )$ de una partícula masiva en reposo relacional es $( \, m_o \, c^2 \, )$

\par \bigskip {\centering\subsubsection*{Magnitudes Relacionales}}

\bigskip \smallskip

\noindent A partir de una partícula masiva auxiliar (denominada punto-auxiliar) es posible obtener magnitudes cinemáticas (denominadas relacionales) que son invariantes bajo transformaciones entre sistemas de referencia inerciales.
\par \bigskip \smallskip
\noindent Un punto-auxiliar es una partícula masiva arbitraria libre de fuerzas externas (o que la fuerza neta actuando sobre ésta es cero)
\par \bigskip \smallskip
\noindent El tiempo relacional $( \, t \, )$, la posición relacional $( \, \mathbf{r} \, )$, la velocidad relacional $( \, \mathbf{v} \, )$ y la aceleración relacional $( \, \mathbf{a} \, )$ de una partícula (masiva o no masiva) respecto a un sistema de referencia inercial S están dados por:
\par \vspace{+0.21em}
\begin{eqnarray*}
t ~\doteq~ \hspace{+0.24em} \gamma \left ( \mathtt{t} - \frac{\vec{r} \cdot \vec{\varphi}}{c^2} \right )
\end{eqnarray*}
\vspace{-0.45em}
\begin{eqnarray*}
\mathbf{r} ~\doteq~ \hspace{-0.09em} \left [ \; \vec{r} + \frac{\gamma^2}{\gamma + 1} \frac{( \vec{r} \cdot \vec{\varphi} \hspace{+0.09em}) \, \vec{\varphi}}{c^2} - \gamma \, \vec{\varphi} \; \mathtt{t} \; \right ]
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathbf{v} \hspace{-0.12em} ~\doteq~ \hspace{-0.12em} \left [ \; \vec{v} + \frac{\gamma^2}{\gamma + 1} \frac{( \vec{v} \cdot \vec{\varphi} \hspace{+0.09em}) \, \vec{\varphi}}{c^2} - \hspace{+0.114em} \gamma \, \vec{\varphi} \hspace{+0.114em} \; \right ] \frac{1}{\gamma \, ( 1 - \frac{\vec{v} \cdot \vec{\varphi}}{c^2} )}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\mathbf{a} ~\doteq~ \hspace{-0.18em} \left [ \; \vec{a} - \frac{\gamma}{\gamma + 1} \frac{( \vec{a} \cdot \vec{\varphi} \hspace{+0.09em}) \, \vec{\varphi}}{c^2} + \frac{( \vec{a} \times \vec{v} \hspace{+0.09em}) \times \vec{\varphi}}{c^2} \; \right ] \frac{1}{\gamma^2 \, ( 1 - \frac{\vec{v} \cdot \vec{\varphi}}{c^2} )^3}
\end{eqnarray*}
\par \vspace{+1.20em}
\noindent donde $( \, \mathtt{t}, \, \vec{r}, \, \vec{v}, \, \vec{a} \, )$ son el tiempo, la posición, la velocidad y la aceleración de la partícula respecto al sistema de referencia inercial S, $( \, \vec{\varphi} \, )$ es la velocidad del punto-auxiliar respecto al sistema de referencia inercial S y $( \, c \, )$ es la velocidad de la luz en el vacío. $( \, \vec{\varphi} \, )$ es una constante. {\small $\gamma = ({1 - \vec{\varphi} \cdot \vec{\varphi}/c^2})^{-1/2}$}

\par \bigskip \smallskip

\noindent La frecuencia relacional $( \: \nu \: )$ de una partícula no masiva respecto a un sistema \hbox {de referencia inercial} S está dada por:
\par \vspace{-0.60em}
\begin{eqnarray*}
\nu ~\doteq~ \mathtt{v} \;\, \dfrac{\left ( 1 - \dfrac{\vec{c} \cdot \vec{\varphi}}{c^2} \right )}{\sqrt{1 - \dfrac{\vec{\varphi} \cdot \vec{\varphi}}{c^2}}}
\end{eqnarray*}
\par \vspace{+0.45em}
\noindent donde $( \: \mathtt{v} \: )$ es la frecuencia de la partícula no masiva respecto al sistema de referencia inercial S, $( \, \vec{c} \, )$ es la velocidad de la partícula no masiva respecto al sistema de referencia inercial S, $( \, \vec{\varphi} \, )$ es la velocidad del punto-auxiliar respecto al sistema de referencia inercial S y $( \, c \, )$ es la velocidad de la luz en el vacío.

\newpage

\noindent § En los sistemas de referencia inerciales no coincidentes \hbox {{\small ( $t_{\alpha} \ne \tau_{\alpha}$\hspace{+0.15em} {\footnotesize y/o} \hspace{+0.18em}$\mathbf{r}_{\alpha} \ne 0$ )}} \hbox {{\small ( $\alpha$ = punto-auxiliar )}} una constante debe ser sumada en la definición de tiempo \hbox {relacional} tal que el tiempo relacional y el tiempo propio del punto-auxiliar sean iguales \hbox {{\small ( $t_{\alpha} = \tau_{\alpha}$ )}} y otra constante debe ser sumada en la definición de posición relacional tal que la posición relacional del punto-auxiliar sea cero \hbox {{\small ( $\mathbf{r}_{\alpha} = 0$ )}}
\par \bigskip \smallskip
\noindent § En el estudio de un sistema aislado de partículas ( masivas {\small y/o} no masivas ) \hbox {los observadores} inerciales deberían preferentemente usar un punto-auxiliar tal que el momento lineal del sistema aislado de partículas sea cero \hbox {{\small ( $\sum_z m_z \hspace{+0.09em} \bar{\mathbf{v}}_z = 0$ )}}

\vspace{-0.60em}

\par \bigskip {\centering\subsubsection*{Observaciones Generales}}

\bigskip \smallskip

\noindent § Las fuerzas y los campos deben ser expresados con magnitudes relacionales \hbox {( la fuerza} de Lorentz debe ser expresada con la velocidad relacional {\small $\mathbf{v}$}, el campo eléctrico debe ser expresado con la posición relacional {\small $\mathbf{r}$}, etc. )
\par \bigskip \smallskip
\noindent § El operador {\small $( \, \dot{\times} \, )$} debe ser reemplazado por el operador {\small $( \, \times \, )$} o el operador {\small $( \, \wedge \, )$} tal como sigue: {\small $( \, \mathbf{a} \hspace{+0.24em}\dot{\times}\hspace{+0.30em} \mathbf{b} = \mathbf{b} \times \mathbf{a} \, )$} o {\small $( \, \mathbf{a} \hspace{+0.24em}\dot{\times}\hspace{+0.30em} \mathbf{b} = \mathbf{b} \wedge \mathbf{a} \, )$}
\par \bigskip \smallskip
\noindent § La magnitud masa intrínseca {\small $( \: m \: )$} es invariante bajo transformaciones entre \hbox {sistemas} de referencia inerciales y no inerciales.
\par \bigskip \smallskip
\noindent § Las magnitudes relacionales {\small $( \; \nu, t, \mathbf{r}, \mathbf{v}, \mathbf{a} \; )$} son invariantes bajo \hbox {transformaciones} entre sistemas de referencia inerciales.
\par \bigskip \smallskip
\noindent § Por lo tanto, las magnitudes cinemáticas y dinámicas {\small $( \: f, \bar{\mathbf{r}}, \bar{\mathbf{v}}, \bar{\mathbf{a}}, \mathbf{P}, \mathbf{L}, \mathbf{F}, \mathrm{W}, \mathrm{K} \: )$} son invariantes bajo transformaciones entre sistemas de referencia inerciales.
\par \bigskip \smallskip
\noindent § Sin embargo, es natural considerar la siguiente generalización:
\par \bigskip \smallskip
\noindent $\bullet$ Sería también posible obtener magnitudes relacionales {\small $( \; \nu, t, \mathbf{r}, \mathbf{v}, \mathbf{a} \; )$} que serían invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.
\par \bigskip \smallskip
\noindent $\bullet$ Las magnitudes cinemáticas y dinámicas {\small $( \: f, \bar{\mathbf{r}}, \bar{\mathbf{v}}, \bar{\mathbf{a}}, \mathbf{P}, \mathbf{L}, \mathbf{F}, \mathrm{W}, \mathrm{K} \: )$} estarían \hbox {dadas} también por las ecuaciones de este artículo.
\par \bigskip \smallskip
\noindent $\bullet$ Por lo tanto, las magnitudes cinemáticas y dinámicas {\small $( \: f, \bar{\mathbf{r}}, \bar{\mathbf{v}}, \bar{\mathbf{a}}, \mathbf{P}, \mathbf{L}, \mathbf{F}, \mathrm{W}, \mathrm{K} \: )$} serían invariantes bajo transformaciones entre sistemas de referencia inerciales y no inerciales.

\par \bigskip {\centering\subsubsection*{Transformaciones Vectoriales de Lorentz}}

\bigskip \smallskip

\noindent Sean dos sistemas de referencia inerciales S y S' cuyos orígenes coinciden en el tiempo cero ( para ambos sistemas ) entonces el tiempo $( \, \mathtt{t}\hspace{+0.03em}' \, )$, la posición $( \, \vec{r}\hspace{+0.18em}' \, )$ la velocidad $( \, \vec{v}\hspace{+0.18em}' \, )$ y la aceleración $( \, \vec{a}\hspace{+0.18em}' \, )$ de una partícula (masiva o no masiva) respecto al sistema de referencia inercial S' están dados por:
\par \vspace{+0.21em}
\begin{eqnarray*}
\mathtt{t}\hspace{+0.03em}' ~=~ \hspace{+0.24em} \gamma \left ( \mathtt{t} - \frac{\vec{r} \cdot \vec{\varphi}}{c^2} \right )
\end{eqnarray*}
\vspace{-0.45em}
\begin{eqnarray*}
\vec{r}\hspace{+0.21em}' ~=~ \hspace{-0.09em} \left [ \; \vec{r} + \frac{\gamma^2}{\gamma + 1} \frac{( \vec{r} \cdot \vec{\varphi} \hspace{+0.09em}) \, \vec{\varphi}}{c^2} - \gamma \, \vec{\varphi} \; \mathtt{t} \; \right ]
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\vec{v}\hspace{+0.21em}' \hspace{-0.12em} ~=~ \hspace{-0.12em} \left [ \; \vec{v} + \frac{\gamma^2}{\gamma + 1} \frac{( \vec{v} \cdot \vec{\varphi} \hspace{+0.09em}) \, \vec{\varphi}}{c^2} - \hspace{+0.114em} \gamma \, \vec{\varphi} \hspace{+0.114em} \; \right ] \frac{1}{\gamma \, ( 1 - \frac{\vec{v} \cdot \vec{\varphi}}{c^2} )}
\end{eqnarray*}
\vspace{-0.30em}
\begin{eqnarray*}
\vec{a}\hspace{+0.21em}' ~=~ \hspace{-0.18em} \left [ \; \vec{a} - \frac{\gamma}{\gamma + 1} \frac{( \vec{a} \cdot \vec{\varphi} \hspace{+0.09em}) \, \vec{\varphi}}{c^2} + \frac{( \vec{a} \times \vec{v} \hspace{+0.09em}) \times \vec{\varphi}}{c^2} \; \right ] \frac{1}{\gamma^2 \, ( 1 - \frac{\vec{v} \cdot \vec{\varphi}}{c^2} )^3}
\end{eqnarray*}
\par \vspace{+1.20em}
\noindent donde $( \, \mathtt{t}, \, \vec{r}, \, \vec{v}, \, \vec{a} \, )$ son el tiempo, la posición, la velocidad y la aceleración de la partícula respecto al sistema de referencia inercial S, $( \, \vec{\varphi} \, )$ es la velocidad del sistema de referencia inercial S' respecto al sistema de referencia inercial S y $( \, c \, )$ es la velocidad de la luz en el vacío. \hspace{-0.09em}$( \, \vec{\varphi} \, )$ es una constante. \hbox {{\small $\gamma = ({1 - \vec{\varphi} \hspace{-0.114em}\cdot\hspace{-0.114em} \vec{\varphi}/c^2})^{-1/2}$}}

\vspace{+0.60em}

\par \bigskip {\centering\subsubsection*{Transformación de Frecuencia}}

\bigskip \smallskip

\noindent La frecuencia $(\hspace{+0.24em}\mathtt{v}\hspace{+0.15em}'\hspace{+0.12em})$ de una partícula no masiva respecto a un sistema de referencia inercial S' está dada por:
\par \vspace{-0.60em}
\begin{eqnarray*}
\mathtt{v}\hspace{+0.15em}' ~=~ \: \mathtt{v} \;\, \dfrac{\left ( 1 - \dfrac{\vec{c} \cdot \vec{\varphi}}{c^2} \right )}{\sqrt{1 - \dfrac{\vec{\varphi} \cdot \vec{\varphi}}{c^2}}}
\end{eqnarray*}
\par \vspace{+0.45em}
\noindent donde $( \: \mathtt{v} \: )$ es la frecuencia de la partícula no masiva respecto a un sistema de referencia inercial S, $( \, \vec{c} \, )$ es la velocidad de la partícula no masiva respecto al sistema de referencia inercial S, $( \, \vec{\varphi} \, )$ es la velocidad del sistema de referencia inercial S' respecto al sistema de referencia inercial S y $( \, c \, )$ es la velocidad de la luz en el vacío.

\par \bigskip {\centering\subsubsection*{Fuerza Cinética}}

\bigskip \smallskip

\noindent La fuerza cinética $\mathbf{K}^{a}_{\hspace{+0.012em}ij}$ ejercida sobre una partícula $i$ con masa intrínseca $m_i$ por otra partícula $j$ con masa intrínseca $m_j$ está dada por:
\par \vspace{-0.54em}
\begin{eqnarray*}
\mathbf{K}^{a}_{\hspace{+0.012em}ij} ~=\, - \; \Bigg [ \; \dfrac{m_i \, m_j}{\mathbb{M}} \, ( \, \bar{\mathbf{a}}_{\hspace{+0.045em}i} - \bar{\mathbf{a}}_{j} \, ) \; \Bigg ]
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent donde $\bar{\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración especial de la partícula $i$, $\bar{\mathbf{a}}_{j}$ es la aceleración especial de la partícula $j$ y $\mathbb{M}$ {\small ( $ = \sum_z m_z$ )} es la suma de las masas intrínsecas de todas las partículas del Universo.
\par \vspace{+0.60em}
\noindent La fuerza cinética $\mathbf{K}^{u}_{\hspace{+0.030em}i}$ ejercida sobre una partícula $i$ con masa intrínseca $m_i$ por el Universo está dada por:
\par \vspace{-0.45em}
\begin{eqnarray*}
\mathbf{K}^{u}_{\hspace{+0.030em}i} ~=\, - \; m_i \; \dfrac{\sum_z m_z \, \bar{\mathbf{a}}_{\hspace{+0.045em}z}}{\sum_z m_z}
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent donde $m_z$ y $\bar{\mathbf{a}}_{\hspace{+0.045em}z}$ son la masa intrínseca y la aceleración especial de la \textit{z}-ésima partícula del Universo.
\par \vspace{+0.60em}
\noindent De las ecuaciones anteriores se deduce que la fuerza cinética neta $\mathbf{K}_i$ {\small ( $ = \sum_j \, \mathbf{K}^{a}_{\hspace{+0.012em}ij}$ $+ \; \mathbf{K}^{u}_{\hspace{+0.030em}i}$ )} que actúa sobre una partícula $i$ con masa intrínseca $m_i$ está dada por:
\par \vspace{-0.60em}
\begin{eqnarray*}
\mathbf{K}_i ~=\, - \; m_i \, \bar{\mathbf{a}}_{\hspace{+0.045em}i}
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent donde $\bar{\mathbf{a}}_{\hspace{+0.045em}i}$ es la aceleración especial de la partícula $i$.
\par \vspace{+0.60em}
\noindent Ahora, reemplazando ( $\mathbf{F}_i = m_i \, \bar{\mathbf{a}}_{\hspace{+0.045em}i}$ ) y reordenando, se obtiene:
\par \vspace{-0.81em}
\begin{eqnarray*}
\mathbf{T}_i ~\doteq~ \mathbf{K}_i + \mathbf{F}_i ~=~ 0
\end{eqnarray*}
\par \vspace{+0.60em}
\noindent Por lo tanto, la fuerza total $\mathbf{T}_i$ que actúa sobre una partícula $i$ es siempre cero.
